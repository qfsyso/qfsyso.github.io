Owntone#

Owntone（以前叫 forked-daapd）是一个自建音乐服务器，支持：

音乐库管理（本地音乐）， AirPlay / Chromecast / Roon / MPD 等协议， 播放列表 / 电台 / 播放控制， 通过 REST API 控制。

TTS（文本转语音）#

用外部TTS 如 Google TTS ， Edge TTS， OpenAI TTS生成.wav .mp3， 放入媒体库文件夹 ，通过API刷新 添加 播放。

Owntone创建#

# 确保有写权限：
sudo chown -R 1000:1000 /var/cache/owntone
sudo chmod -R 755 /var/cache/owntone


docker run -d \
  --name=OwnTone1 \
  --network=host \
  -e UID=1000 \
  -e GID=1000 \
  -v /etc/owntone:/etc/owntone \
  -v /mnt/media:/srv/media \
  -v /var/cache/owntone:/var/cache/owntone \
  --restart unless-stopped \
  owntone/owntone:latest

# 停止删除
docker stop OwnTone1 && docker rm OwnTone1
# 精简版
docker run -d \
  --name=OwnTone1 \
  --network=host \
  -v /root/owntone/config:/etc/owntone \
  -v /root/owntone/media:/srv/media/ \
  -v /root/owntone/cache:/var/cache/owntone \
  --restart unless-stopped \
  docker.io/owntone/owntone:28.12

gTTS#

app.py


import os
import time
from gtts import gTTS
import requests

# === 配置区 ===
OWNTONE_URL = "http://127.0.0.1:3689"   # Owntone 的 API 地址
MEDIA_PATH = "/mnt/media"                # 你的宿主机挂载目录
TTS_FILENAME = "tts_output.mp3"         # TTS 文件名

# === 1) 生成 TTS ===
text = "你好，我是 Owntone 的 TTS 播放示例"
tts = gTTS(text, lang="zh-CN")
tts_file_path = os.path.join(MEDIA_PATH, TTS_FILENAME)
tts.save(tts_file_path)
print(f"✅ 已生成 TTS 文件：{tts_file_path}")

# === 2) 刷新媒体库 ===
print("🔄 刷新 Owntone 媒体库...")
try:
    resp = requests.put(f"{OWNTONE_URL}/api/rescan")
    if resp.status_code == 200:
        print("✅ 已请求刷新")
    else:
        print(f"⚠️ 刷新失败: {resp.status_code} {resp.text}")
except Exception as e:
    print(f"❌ 刷新异常: {e}")

# === 3) 等待扫描完成 ===
print("⏳ 等待 5 秒以保证文件被扫描...")
time.sleep(5)

# === 4) 搜索 TTS 文件 ===
print("🔍 搜索 TTS 文件...")
try:
    search_resp = requests.get(f"{OWNTONE_URL}/api/search?query={TTS_FILENAME}")
    if search_resp.status_code != 200:
        print(f"❌ 搜索失败: {search_resp.status_code} {search_resp.text}")
        exit(1)

    search_json = search_resp.json()
    tracks = search_json.get("tracks", [])

    if not tracks:
        print(f"❌ 没找到 TTS 文件，API 返回: {search_json}")
        exit(1)

    track_id = tracks[0]["id"]
    print(f"✅ 找到 TTS 文件，ID: {track_id}")

except Exception as e:
    print(f"❌ 搜索异常: {e}")
    print(f"搜索接口返回: {search_resp.text}")
    exit(1)

# === 5) 加到队列 ===
print("➕ 加入播放队列...")
add_resp = requests.post(
    f"{OWNTONE_URL}/api/queue/items/add",
    json={"uris": [f"library:track:{track_id}"]}
)
if add_resp.status_code == 200:
    print("✅ 已加入队列")
else:
    print(f"⚠️ 加队列失败: {add_resp.status_code} {add_resp.text}")

# === 6) 播放 ===
print("▶️ 播放...")
play_resp = requests.put(f"{OWNTONE_URL}/api/player/play")
if play_resp.status_code == 200:
    print("✅ 已开始播放")
else:
    print(f"⚠️ 播放失败: {play_resp.status_code} {play_resp.text}")

要点#

PUT /api/rescan 代替了老的 POST /api/library/refresh， 增加了对返回码和异常的判断， 如果搜索失败会打印返回内容，方便调试。

如果需要用户名密码，要把 URL 写成：#

OWNTONE_URL = "http://username:password@127.0.0.1:3689"

import os
import sys
import time
from datetime import datetime
from gtts import gTTS

OWNTONE_URL = "http://192.168.80.39:3689"  # Owntone 的 API 地址
MEDIA_PATH = "/mnt/media"                  # 宿主机挂载到 Owntone 的目录
LANG = "zh-CN"                             # TTS 语言

# === 1) 获取要读的文字 ===
if len(sys.argv) < 2:
    print("❗ 用法: python tts_play.py \"要朗读的文字\"")
    sys.exit(1)

text = sys.argv[1]
timestamp = datetime.now().strftime("%Y%m%d%H%M%S")
tts_filename = f"tts_{timestamp}.mp3"
tts_file_path = os.path.join(MEDIA_PATH, tts_filename)

    else:
        print(f"⚠️ 刷新失败: {resp.status_code} {resp.text}")

要点#

自动带时间戳，防止覆盖， 完整流程：生成 → 刷库 → 搜索 → 加队列 → 播放， 失败时退出码 ≠ 0，方便做定时任务或外部调用检测， 这样就一条命令从生成到播放。

edge-tts + Owntone#


#!/usr/bin/env python3
import os
import sys
import time
import asyncio
from datetime import datetime
import requests
import edge_tts  # 用的是异步版

# === 配置 ===
OWNTONE_URL = "http://192.168.80.39:3689"  # Owntone API
MEDIA_PATH = "/mnt/media"                  # Owntone 宿主机挂载目录
VOICE = "zh-CN-XiaoxiaoNeural"             # 微软 Edge TTS 的中文自然女声

# === 检查参数 ===
if len(sys.argv) < 2:
    print("❗ 用法: python tts_play_edge.py \"要朗读的文字\"")
    sys.exit(1)

text = sys.argv[1]
timestamp = datetime.now().strftime("%Y%m%d%H%M%S")
tts_filename = f"tts_{timestamp}.mp3"
tts_file_path = os.path.join(MEDIA_PATH, tts_filename)


async def main():
    # === 1) 使用 edge-tts 生成 TTS ===
    print(f"🗣️ 正在用 Edge-TTS 生成：{tts_file_path}")
    communicate = edge_tts.Communicate(text, VOICE)
    await communicate.save(tts_file_path)
    print(f"✅ 已生成 TTS 文件: {tts_file_path}")

    # === 2) 刷 Owntone 库 ===
    print("🔄 正在刷新 Owntone 媒体库...")
    try:
        resp = requests.put(f"{OWNTONE_URL}/api/rescan")
        if resp.status_code == 204:
            print("✅ 已请求刷新")
        else:
            print(f"⚠️ 刷新失败: {resp.status_code} {resp.text}")
            sys.exit(1)
    except Exception as e:
        print(f"❌ 刷新异常: {e}")
        sys.exit(1)

    # === 3) 等待 ===
    print("⏳ 等待 5 秒以保证文件被扫描...")
    time.sleep(5)

    # === 4) 搜索文件 ===
    print("🔍 正在搜索 TTS 文件...")
    try:
        params = {"query": tts_filename, "type": "track"}
        search_resp = requests.get(f"{OWNTONE_URL}/api/search", params=params)
        if search_resp.status_code != 200:
            print(f"❌ 搜索失败: {search_resp.status_code} {search_resp.text}")
            sys.exit(1)

        search_json = search_resp.json()
        tracks = search_json.get("tracks", {}).get("items", [])
        if not tracks:
            print(f"❌ 没找到 TTS 文件，API 返回: {search_json}")
            sys.exit(1)

        track_id = tracks[0]["id"]
        track_title = tracks[0].get("title", "N/A")
        print(f"✅ 找到 TTS 文件 ID: {track_id}，标题: {track_title}")

    except Exception as e:
        print(f"❌ 搜索异常: {e}")
        sys.exit(1)

    # === 5) 加队列 ===
    print("➕ 正在加入播放队列...")
    try:
        add_url = f"{OWNTONE_URL}/api/queue/items/add"
        add_params = {"uris": f"library:track:{track_id}"}
        add_resp = requests.post(add_url, params=add_params)
        if add_resp.status_code != 200:
            print(f"⚠️ 加队列失败: {add_resp.status_code} {add_resp.text}")
            sys.exit(1)
        print("✅ 已加入播放队列")
    except Exception as e:
        print(f"❌ 加队列异常: {e}")
        sys.exit(1)

    # === 6) 播放 ===
    print("▶️ 正在播放...")
    try:
        play_url = f"{OWNTONE_URL}/api/player/play"
        play_resp = requests.put(play_url)
        if play_resp.status_code != 204:
            print(f"⚠️ 播放失败: {play_resp.status_code} {play_resp.text}")
            sys.exit(1)
        print("✅ 已开始播放 🎉")
    except Exception as e:
        print(f"❌ 播放异常: {e}")
        sys.exit(1)


if __name__ == "__main__":
    asyncio.run(main())

优点#

edge-tts 是微软 Edge 浏览器的官方语音合成，效果比 gTTS（Google Translate）更自然， 可选多种中文/英文语音，支持调节语速/情感， 全流程和 gTTS 版一致，一条命令就能跑。

声音选择#

tts_play.py

import os
import sys
import time
import asyncio
from datetime import datetime

import requests
import edge_tts

# === 配置 ===
OWNTONE_URL = "http://192.168.80.39:3689"  # Owntone API
MEDIA_PATH = "/mnt/media"  # 宿主机挂载的媒体库路径

# 可选的中文声音
VOICES = {
    "xiaoxiao": "zh-CN-XiaoxiaoNeural",  # 普通女声
    "xiaoyi": "zh-CN-XiaoyiNeural",      # 活泼女声
    "yunxi": "zh-CN-YunxiNeural",        # 稳重男声
    "yunjian": "zh-CN-YunjianNeural"     # 男声
}


async def main():
    # === 解析命令行 ===
    if len(sys.argv) < 2:
        print(f"用法: python tts_play.py \"要读的文字\" [voice]\n可选 voice: {', '.join(VOICES.keys())}")
        sys.exit(1)

    text = " ".join(sys.argv[1:-1]) if len(sys.argv) > 2 and sys.argv[-1].lower() in VOICES else " ".join(sys.argv[1:])
    voice_key = sys.argv[-1].lower() if sys.argv[-1].lower() in VOICES else "xiaoxiao"
    voice = VOICES[voice_key]

    print(f"🎙️ 文字: {text}")
    print(f"🔈 声音: {voice} ({voice_key})")

    # === 1) 生成 TTS ===
    timestamp = datetime.now().strftime("%Y%m%d%H%M%S")
    tts_filename = f"tts_{timestamp}.mp3"
    tts_file_path = os.path.join(MEDIA_PATH, tts_filename)

    print("⏳ 正在生成 TTS...")
    communicate = edge_tts.Communicate(text, voice)
    await communicate.save(tts_file_path)
    print(f"✅ 已生成: {tts_file_path}")

    # === 2) Owntone 刷库 ===
    print("🔄 刷新 Owntone 媒体库...")
    try:
        resp = requests.put(f"{OWNTONE_URL}/api/rescan")
        if resp.status_code == 204:
            print("✅ 已请求刷新")
        else:
            print(f"⚠️ 刷新失败: {resp.status_code} {resp.text}")
            sys.exit(1)
    except Exception as e:
        print(f"❌ 刷新异常: {e}")
        sys.exit(1)

    # === 3) 等待扫描 ===
    print("⏳ 等待 5 秒以保证文件被扫描...")
    time.sleep(5)

    # === 4) 搜索文件 ===
    print("🔍 搜索 TTS 文件...")
    try:
        params = {"query": tts_filename, "type": "track"}
        search_resp = requests.get(f"{OWNTONE_URL}/api/search", params=params)
        if search_resp.status_code != 200:
            print(f"❌ 搜索失败: {search_resp.status_code} {search_resp.text}")
            sys.exit(1)

        search_json = search_resp.json()
        tracks = search_json.get("tracks", {}).get("items", [])
        if not tracks:
            print(f"❌ 没找到文件，API 返回: {search_json}")
            sys.exit(1)

        track_id = tracks[0]["id"]
        print(f"✅ 找到 TTS 文件，ID: {track_id}")

    except Exception as e:
        print(f"❌ 搜索异常: {e}")
        sys.exit(1)

    # === 5) 加入播放队列 ===

# 可选的中文声音
VOICES = {
    "xiaoxiao": "zh-CN-XiaoxiaoNeural",
    "xiaoyi": "zh-CN-XiaoyiNeural",
    "yunxi": "zh-CN-YunxiNeural",
    "yunjian": "zh-CN-YunjianNeural"
}


async def handle_tts(update: Update, context: ContextTypes.DEFAULT_TYPE):
    args = context.args

    if not args:
        await update.message.reply_text(
            "📝 用法: /tts <要读的文字> [voice]\n"
            "例如: /tts 你好 世界 xiaoyi\n\n"
            f"可选 voice: {', '.join(VOICES.keys())}"
        )
        return

    text = " ".join(args[:-1]) if len(args) > 1 and args[-1].lower() in VOICES else " ".join(args)
    voice_key = args[-1].lower() if args[-1].lower() in VOICES else "xiaoxiao"
    voice = VOICES[voice_key]

    # 文件名带时间戳
    timestamp = datetime.now().strftime("%Y%m%d%H%M%S")
    tts_filename = f"tts_{timestamp}.mp3"
    tts_file_path = os.path.join(MEDIA_PATH, tts_filename)

    await update.message.reply_text(f"🎙️ 正在生成 TTS（Voice: {voice_key}）...")

    # === 1) 用 edge-tts 生成 ===
    try:
        communicate = edge_tts.Communicate(text, voice)
        await communicate.save(tts_file_path)
        await update.message.reply_text(f"✅ 已生成: {tts_filename}")
    except Exception as e:
        await update.message.reply_text(f"❌ TTS 生成失败: {e}")
        return

    # === 2) Owntone 刷库 ===
    await update.message.reply_text("🔄 正在刷新 Owntone 媒体库...")
    try:
        resp = requests.put(f"{OWNTONE_URL}/api/rescan")
        if resp.status_code == 204:
            await update.message.reply_text("✅ 已请求刷新")
        else:
            await update.message.reply_text(f"⚠️ 刷新失败: {resp.status_code} {resp.text}")
            return
    except Exception as e:
        await update.message.reply_text(f"❌ 刷新异常: {e}")
        return

    # === 3) 等待 ===
    await update.message.reply_text("⏳ 等待 5 秒以保证文件被扫描...")
    time.sleep(5)

    # === 4) 搜索文件 ===
    try:
        params = {"query": tts_filename, "type": "track"}
        search_resp = requests.get(f"{OWNTONE_URL}/api/search", params=params)
        if search_resp.status_code != 200:
            await update.message.reply_text(f"❌ 搜索失败: {search_resp.status_code} {search_resp.text}")
            return

        search_json = search_resp.json()
        tracks = search_json.get("tracks", {}).get("items", [])
        if not tracks:
            await update.message.reply_text(f"❌ 没找到文件，API 返回: {search_json}")
            return

        track_id = tracks[0]["id"]
    except Exception as e:
        await update.message.reply_text(f"❌ 搜索异常: {e}")
        return

    # === 5) 加入播放队列 ===
    try:
        add_url = f"{OWNTONE_URL}/api/queue/items/add"
        add_params = {"uris": f"library:track:{track_id}"}
        add_resp = requests.post(add_url, params=add_params)
        if add_resp.status_code != 200:
            await update.message.reply_text(f"⚠️ 加队列失败: {add_resp.status_code} {add_resp.text}")
            return
        await update.message.reply_text("✅ 已加入播放队列")
    except Exception as e:
        await update.message.reply_text(f"❌ 加队列异常: {e}")
        return

    # === 6) 播放 ===
    try:
        play_url = f"{OWNTONE_URL}/api/player/play"
        play_resp = requests.put(play_url)
        if play_resp.status_code != 204:
            await update.message.reply_text(f"⚠️ 播放失败: {play_resp.status_code} {play_resp.text}")
            return
        await update.message.reply_text("▶️ 正在播放！🎉")
    except Exception as e:
        await update.message.reply_text(f"❌ 播放异常: {e}")

import os
import asyncio
import time
from datetime import datetime

import requests
import edge_tts
from telegram import Update
from telegram.ext import ApplicationBuilder, CommandHandler, ContextTypes

# === 配置 ===
OWNTONE_URL = "http://192.168.80.39:3689"
MEDIA_PATH = "/mnt/media"

VOICES = {
    "xiaoxiao": "zh-CN-XiaoxiaoNeural",
    "xiaoyi": "zh-CN-XiaoyiNeural",
    "yunxi": "zh-CN-YunxiNeural",
    "yunjian": "zh-CN-YunjianNeural"
}


async def generate_tts(text, voice_key):
    voice = VOICES.get(voice_key, VOICES["xiaoxiao"])
    timestamp = datetime.now().strftime("%Y%m%d%H%M%S")
    tts_filename = f"tts_{timestamp}.mp3"
    tts_file_path = os.path.join(MEDIA_PATH, tts_filename)

    communicate = edge_tts.Communicate(text, voice)
    await communicate.save(tts_file_path)
    return tts_filename


def owntone_rescan():
    resp = requests.put(f"{OWNTONE_URL}/api/rescan")
    if resp.status_code == 204:
        return True
    else:
        return False


def owntone_search(filename):
    params = {"query": filename, "type": "track"}
    resp = requests.get(f"{OWNTONE_URL}/api/search", params=params)
    if resp.status_code != 200:
        return None

    tracks = resp.json().get("tracks", {}).get("items", [])
    if not tracks:
        return None

    return tracks[0]["id"]


def owntone_add_queue(track_id):
    url = f"{OWNTONE_URL}/api/queue/items/add"
    params = {"uris": f"library:track:{track_id}"}
    resp = requests.post(url, params=params)
    return resp.status_code == 200


def owntone_play():
    resp = requests.put(f"{OWNTONE_URL}/api/player/play")
    return resp.status_code == 204


async def tts_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    args = context.args
    if not args:
        await update.message.reply_text("用法: /tts <内容> [声音: xiaoxiao|xiaoyi|yunxi|yunjian]")
        return

    text = " ".join(args[:-1]) if args[-1] in VOICES else " ".join(args)
    voice_key = args[-1] if args[-1] in VOICES else "xiaoxiao"

    await update.message.reply_text(f"生成中... 声音: {voice_key}")

    tts_filename = await generate_tts(text, voice_key)
    await update.message.reply_text(f"TTS 文件已生成: {tts_filename}")

    if not owntone_rescan():
        await update.message.reply_text("Owntone 刷库失败！")
        return

    await update.message.reply_text("等待 Owntone 刷库...")
    time.sleep(5)

    track_id = owntone_search(tts_filename)
    if not track_id:
        await update.message.reply_text("没找到生成的音频文件！")
        return

    if not owntone_add_queue(track_id):
        await update.message.reply_text("加入播放队列失败！")
        return

    if not owntone_play():
        await update.message.reply_text("播放失败！")
        return

    await update.message.reply_text("🎉 已开始播放！")


async def main():
    token = os.getenv("TELEGRAM_BOT_TOKEN")
    if not token:
        print("❌ TELEGRAM_BOT_TOKEN 未设置")
        return

    app = ApplicationBuilder().token(token).build()
    app.add_handler(CommandHandler("tts", tts_command))
    print("✅ 机器人已启动")
    await app.run_polling()


if __name__ == "__main__":
    asyncio.run(main())
