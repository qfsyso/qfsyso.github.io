Dcoker Bakup#

导出时生成 .tar + _config.json，记录容器端口、卷挂载和环境变量。

导入时自动读取配置文件，本地导入或远程导入都能恢复原容器配置。

支持本地导入：无需输入服务器信息。

支持远程导入：传输 .tar 和 .json 到服务器，并在服务器上启动容器。

Code#

python3 -m venv myenv
source myenv/bin/activate


import os
import json
import subprocess
import paramiko

def list_containers():
    result = subprocess.run(["docker", "ps", "--format", "{{.Names}}"], capture_output=True, text=True)
    containers = result.stdout.strip().split("\n")
    return containers

def inspect_container(name):
    result = subprocess.run(["docker", "inspect", name], capture_output=True, text=True)
    info = json.loads(result.stdout)[0]
    ports = info['HostConfig']['PortBindings']
    volumes = info['Mounts']
    envs = info['Config']['Env']
    return ports, volumes, envs

def export_container(name, image_name):
    subprocess.run(["docker", "commit", name, image_name], check=True)
    tar_file = f"{image_name}.tar"
    subprocess.run(["docker", "save", "-o", tar_file, image_name], check=True)
    return tar_file

def save_container_config(name, ports, volumes, envs):
    config_file = f"{name}_image_config.json"
    config = {"ports": ports, "volumes": volumes, "envs": envs}
    with open(config_file, "w") as f:
        json.dump(config, f, indent=4)
    print(f"容器配置已保存为 {config_file}")
    return config_file

def transfer_file(local_file, target_host, target_user, target_path):
    ssh = paramiko.SSHClient()
    ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
    ssh.connect(target_host, username=target_user)
    sftp = ssh.open_sftp()
    sftp.put(local_file, os.path.join(target_path, os.path.basename(local_file)))
    sftp.close()
    ssh.close()

def restore_container(tar_file, container_name, ports, volumes, envs, local=True, target_host=None, target_user=None, target_path=None):
    port_args = " ".join([f"-p {v[0]['HostPort']}:{k.split('/')[0]}" for k,v in ports.items()])
    volume_args = " ".join([f"-v {v['Source']}:{v['Destination']}" for v in volumes])
    env_args = " ".join([f"-e {e}" for e in envs])
    image_name = tar_file.replace('.tar','')
    run_cmd = f"docker run -d --name {container_name} {port_args} {volume_args} {env_args} {image_name}"

    if local:
        subprocess.run(f"docker load -i {tar_file}", shell=True, check=True)
        subprocess.run(run_cmd, shell=True, check=True)
        print(f"容器 {container_name} 已在本地启动。")
    else:
        ssh = paramiko.SSHClient()
        ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
        ssh.connect(target_host, username=target_user)
        ssh.exec_command(f"docker load -i {os.path.join(target_path, os.path.basename(tar_file))}")
        ssh.exec_command(run_cmd)
        ssh.close()
        print(f"容器 {container_name} 已在远程服务器 {target_host} 启动。")

def export_flow():
    containers = list_containers()
    print("本地运行容器列表：")
    for idx, c in enumerate(containers):
        print(f"{idx}: {c}")
    choice = int(input("选择要导出的容器编号: "))
    container_name = containers[choice]

    ports, volumes, envs = inspect_container(container_name)
    print("导出镜像中...")
    tar_file = export_container(container_name, container_name+"_image")
    print(f"镜像已导出为 {tar_file}")

    config_file = save_container_config(container_name, ports, volumes, envs)

    do_transfer = input("是否传输到目标服务器？(y/n): ").strip().lower()
    if do_transfer == 'y':
        target_host = input("目标服务器IP: ")
        target_user = input("目标服务器用户名: ")
        target_path = input("目标服务器存放路径: ")

        print("传输镜像和配置文件中...")
        transfer_file(tar_file, target_host, target_user, target_path)
        transfer_file(config_file, target_host, target_user, target_path)

        print("在目标服务器恢复容器...")
        restore_container(tar_file, container_name, ports, volumes, envs, local=False,
                          target_host=target_host, target_user=target_user, target_path=target_path)
        print("迁移完成！")
    else:
        print("镜像和配置文件已保存到本地，未传输到服务器。")

def import_flow():
    tar_file = input("输入本地镜像文件路径 (.tar): ").strip()
    config_file = tar_file.replace('.tar', '_config.json')
    container_name = input("输入要恢复的容器名称: ").strip()
    mode = input("选择导入模式：1-本地导入  2-远程导入: ").strip()

    if not os.path.exists(config_file):
        print(f"配置文件 {config_file} 不存在，无法恢复端口、卷和环境变量！")
        ports, volumes, envs = {}, [], []
    else:
        with open(config_file, 'r') as f:
            cfg = json.load(f)
        ports, volumes, envs = cfg.get('ports', {}), cfg.get('volumes', []), cfg.get('envs', [])

    if mode == '1':
        restore_container(tar_file, container_name, ports, volumes, envs, local=True)
    elif mode == '2':
        target_host = input("目标服务器IP: ")
        target_user = input("目标服务器用户名: ")
        target_path = input("目标服务器存放路径: ").strip()
        print("传输镜像和配置文件中...")
        transfer_file(tar_file, target_host, target_user, target_path)
        transfer_file(config_file, target_host, target_user, target_path)
        restore_container(tar_file, container_name, ports, volumes, envs, local=False,
                          target_host=target_host, target_user=target_user, target_path=target_path)
    else:
        print("无效选择！")

if __name__ == "__main__":
    print("选择操作：")
    print("1: 导出容器到本地镜像")
    print("2: 将本地镜像导入并恢复容器")
    choice = input("输入 1 或 2: ").strip()

    if choice == '1':
        export_flow()
    elif choice == '2':
        import_flow()
    else:
        print("无效选择！")

import os
import json
import subprocess
import paramiko

def list_containers():
    result = subprocess.run(["docker", "ps", "--format", "{{.Names}}"], capture_output=True, text=True)
    containers = result.stdout.strip().split("\n")
    return containers

def inspect_container(name):
    result = subprocess.run(["docker", "inspect", name], capture_output=True, text=True)
    info = json.loads(result.stdout)[0]

    ports = info['HostConfig'].get('PortBindings', {})
    volumes = info.get('Mounts', [])
    envs = info['Config'].get('Env', [])
    networks = {}
    
    for net_name, net_info in info['NetworkSettings']['Networks'].items():
        networks[net_name] = {
            'Aliases': net_info.get('Aliases', []),
            'IPAddress': net_info.get('IPAddress')  # 可选，Docker 会自动分配
        }

    return ports, volumes, envs, networks

    subprocess.run(["docker", "commit", name, image_name], check=True)
    tar_file = f"{image_name}.tar"

    return tar_file

    config_file = f"{name}_image_config.json"
    config = {
        "ports": ports,
        "volumes": volumes,
        "envs": envs,
        "networks": networks
    }

    return config_file

import os

import paramiko
from getpass import getpass

def list_containers():

    containers = result.stdout.strip().split("\n")

    subprocess.run(["docker", "commit", name, image_name], check=True)
    tar_file = f"{image_name}.tar"
    subprocess.run(["docker", "save", "-o", tar_file, image_name], check=True)
    return tar_file

def save_container_config(name, ports, volumes, envs, networks):
    config_file = f"{name}_image_config.json"
    config = {
        "ports": ports,
        "volumes": volumes,
        "envs": envs,
        "networks": networks
    }
    with open(config_file, "w") as f:
        json.dump(config, f, indent=4)
    print(f"容器配置已保存为 {config_file}")
    return config_file

def transfer_file(local_file, target_host, target_user, target_path, port=22, password=None, key_filename=None):
    ssh = paramiko.SSHClient()
    ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
    ssh.connect(target_host, port=port, username=target_user, password=password, key_filename=key_filename)
    sftp = ssh.open_sftp()
    sftp.put(local_file, os.path.join(target_path, os.path.basename(local_file)))
    sftp.close()
    ssh.close()

def restore_container(tar_file, container_name, ports, volumes, envs, networks, local=True, target_host=None, target_user=None, target_path=None, port=22, password=None, key_filename=None):
    port_args = " ".join([f"-p {v[0]['HostPort']}:{k.split('/')[0]}" for k,v in ports.items()])
    volume_args = " ".join([f"-v {v['Source']}:{v['Destination']}" for v in volumes])
    env_args = " ".join([f"-e {e}" for e in envs])
    
    image_name = os.path.basename(tar_file).replace('.tar','')
    run_cmd = f"docker run -d --name {container_name} {port_args} {volume_args} {env_args} {image_name}"

    if local:
        subprocess.run(f"docker load -i {tar_file}", shell=True, check=True)
        subprocess.run(run_cmd, shell=True, check=True)
        for net_name, net_cfg in networks.items():
            if net_cfg is None:
                continue
            aliases = net_cfg.get('Aliases') or []
            subprocess.run(f"docker network inspect {net_name} || docker network create {net_name}", shell=True)
            for alias in aliases:
                subprocess.run(f"docker network connect --alias {alias} {net_name} {container_name}", shell=True)
        print(f"容器 {container_name} 已在本地启动。")
    else:
        ssh = paramiko.SSHClient()
        ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
        ssh.connect(target_host, port=port, username=target_user, password=password, key_filename=key_filename)
        ssh.exec_command(f"docker load -i {os.path.join(target_path, os.path.basename(tar_file))}")
        ssh.exec_command(run_cmd)
        for net_name, net_cfg in networks.items():
            if net_cfg is None:
                continue
            aliases = net_cfg.get('Aliases') or []
            ssh.exec_command(f"docker network inspect {net_name} || docker network create {net_name}")
            for alias in aliases:
                ssh.exec_command(f"docker network connect --alias {alias} {net_name} {container_name}")
        ssh.close()
        print(f"容器 {container_name} 已在远程服务器 {target_host} 启动。")

def export_flow(batch=False):
    containers = list_containers()
    print("本地运行容器列表：")
    for idx, c in enumerate(containers):
        print(f"{idx}: {c}")

    if batch:
        choices = input("输入要导出的容器编号(多个用逗号隔开，例如0,2,3): ")
        choices = [int(x.strip()) for x in choices.split(",")]
    else:
        choice = int(input("选择要导出的容器编号: "))
        choices = [choice]

    exported = []
    for idx in choices:
        container_name = containers[idx]
        ports, volumes, envs, networks = inspect_container(container_name)
        print(f"导出容器 {container_name} 中...")
        tar_file = export_container(container_name, container_name + "_image")
        print(f"镜像已导出为 {tar_file}")
        config_file = save_container_config(container_name, ports, volumes, envs, networks)
        exported.append((tar_file, config_file, container_name, ports, volumes, envs, networks))

    do_transfer = input("是否传输到目标服务器？(y/n): ").strip().lower()
    if do_transfer == 'y':
        target_host = input("目标服务器IP: ").strip()
        target_port = int(input("目标服务器SSH端口(默认22): ").strip() or 22)
        target_user = input("目标服务器用户名: ").strip()
        auth_method = input("使用方式 1-密码登录 2-私钥登录: ").strip()
        password = None
        key_filename = None
        if auth_method == '1':
            password = getpass("输入目标服务器密码: ")
        elif auth_method == '2':
            key_filename = input("输入私钥路径: ").strip()
        else:
            print("无效认证方式")
            return
        target_path = input("目标服务器存放路径: ").strip()

        for tar_file, config_file, container_name, ports, volumes, envs, networks in exported:
            print(f"传输 {container_name} 镜像和配置文件中...")
            transfer_file(tar_file, target_host, target_user, target_path, port=target_port, password=password, key_filename=key_filename)
            transfer_file(config_file, target_host, target_user, target_path, port=target_port, password=password, key_filename=key_filename)
            print(f"在目标服务器恢复容器 {container_name}...")
            restore_container(tar_file, container_name, ports, volumes, envs, networks, local=False,
                              target_host=target_host, target_user=target_user, target_path=target_path,
                              port=target_port, password=password, key_filename=key_filename)
        print("批量迁移完成！")
    else:
        print("镜像和配置文件已保存到本地，未传输到服务器。")

def import_flow(batch=False):
    if batch:
        dir_path = input("输入包含镜像的目录路径: ").strip()
        if not os.path.isdir(dir_path):
            print("目录不存在！")
            return
        tar_files = [os.path.join(dir_path, f) for f in os.listdir(dir_path) if f.endswith(".tar")]
        if not tar_files:
            print("目录中没有 .tar 文件！")
            return
    else:
        tar_file = input("输入本地镜像文件路径 (.tar): ").strip()
        tar_files = [tar_file]

    mode = input("选择导入模式：1-本地导入  2-远程导入: ").strip()
    password = None
    key_filename = None
    target_port = 22
    if mode == '2':
        target_host = input("目标服务器IP: ").strip()
        target_port = int(input("目标服务器SSH端口(默认22): ").strip() or 22)
        target_user = input("目标服务器用户名: ").strip()
        auth_method = input("使用方式 1-密码登录 2-私钥登录: ").strip()
        if auth_method == '1':
            password = getpass("输入目标服务器密码: ")
        elif auth_method == '2':
            key_filename = input("输入私钥路径: ").strip()
        else:
            print("无效认证方式")
            return
        target_path = input("目标服务器存放路径: ").strip()

    for tar_file in tar_files:
        config_file = tar_file.replace('.tar', '_config.json')
        container_name = os.path.basename(tar_file).replace('_image.tar','')

        if not os.path.exists(config_file):
            print(f"配置文件 {config_file} 不存在，无法恢复端口、卷、环境变量和网络！")
            ports, volumes, envs, networks = {}, [], [], {}
        else:
            with open(config_file, 'r') as f:
                cfg = json.load(f)
            ports, volumes, envs, networks = cfg.get('ports', {}), cfg.get('volumes', []), cfg.get('envs', []), cfg.get('networks', {})

        print(f"恢复容器 {container_name} ...")
        if mode == '1':
            restore_container(tar_file, container_name, ports, volumes, envs, networks, local=True)
        elif mode == '2':
            transfer_file(tar_file, target_host, target_user, target_path, port=target_port, password=password, key_filename=key_filename)
            if os.path.exists(config_file):
                transfer_file(config_file, target_host, target_user, target_path, port=target_port, password=password, key_filename=key_filename)
            restore_container(tar_file, container_name, ports, volumes, envs, networks, local=False,
                              target_host=target_host, target_user=target_user, target_path=target_path,
                              port=target_port, password=password, key_filename=key_filename)
    print("批量导入完成！")
