  <div v-if="scope.row.MsgType == 7 || scope.row.MsgType == 19">
    <el-button
      :type="playingRowId === scope.row.AutoID ? 'warning' : 'info'"
      :disabled="playingRowId === scope.row.AutoID"
      @click="handleD(scope.row.Content, scope.row.MsgType, scope.row.UserID, scope.row.AutoID)"
    >

JS代码


<script setup>
import { ref } from 'vue';
import { ElMessage } from 'element-plus';
import request from '@/api/request'; // Assuming you use Axios or similar

const playingRowId = ref(null); // Tracks the AutoID of the row currently being played
const dialogVisible = ref(false);
const mediaSource = ref('');

async function handleD(com, msgt, userid, autoid) {
  try {
    // Set the current row's AutoID as the playing row
    playingRowId.value = autoid;

    const response = await request({
      url: '/adm/zuserApi/FdMsgJMCon',
      method: 'put',
      data: {
        com,
        msgtype: String(msgt),
        userid: String(userid) // Use UserID for API request
      }
    });

    const { isOk, msg } = response;

    if (isOk) {
      if (msg.endsWith('.aac')) {
        playAAC(msg, autoid);
      } else if (msg.endsWith('.mp4')) {
        playMP4(msg);
      }
    } else {
      ElMessage.error('获取失败，请检查网络连接或联系管理员');
    }
  } catch (error) {
    console.error('获取时发生错误:', error);
    ElMessage.error('获取失败，请检查网络连接或联系管理员');
  }
}

function playAAC(source, autoid) {
  const audio = new Audio(source);

  audio.play();
  audio.onended = () => {
    // Reset the playing row when playback finishes
    if (playingRowId.value === autoid) {
      playingRowId.value = null;
    }
  };

  audio.onerror = () => {
    if (playingRowId.value === autoid) {
      playingRowId.value = null;
    }
    ElMessage.error('音频播放失败');
  };
}

function playMP4(source) {
  mediaSource.value = source;
  dialogVisible.value = true;
}

function handleDialogClose() {
  dialogVisible.value = false;
  playingRowId.value = null; // Reset when the dialog is closed
}
</script>

设置媒体源（mediaSource）：确保在打开对话框之前，视频的来源（mediaSource.value）已正确设置为MP4的URL。 设置对话框可见性（dialogVisible）：必须将dialogVisible.value设置为true，以确保对话框能够打开。 **视频元素：**视频元素的src属性应绑定到mediaSource，并且该元素应在el-dialog组件中可见。

弹窗注意引入 Element Plus v-model=“dialogVisible”

console.log(‘Dialog visible:’, dialogVisible.value); // true console.log(‘Video source:’, mediaSource.value); // MP4 URL

图片显示#

使用 Element Plus 提供的 el-image-viewer 组件。这个组件允许你点击图片时显示大图，带有放大、缩小、旋转等功能

 <!-- 缩略图 -->
    <el-image
      style="width: 80px; height: 80px"
      :src="scope.row.ContentT2"
      @click="handleImageClick(scope.row.ContentT2)"
    >
      <template #error> ERR </template>
    </el-image>

    <!-- 大图查看器 -->
    <el-dialog :visible.sync="dialogVisible" width="80%" center @close="closeDialog" append-to-body>
      <el-image-viewer :url-list="urlList" :initial-index="0" @close="closeDialog" v-if="dialogVisible" />
    </el-dialog>

JS代码

<script setup>
import { ref } from 'vue';

const dialogVisible = ref(false); // 控制对话框是否可见
const urlList = ref([]); // 图片列表

// 点击图片时触发，打开图片查看器
function handleImageClick(imageUrl) {
  urlList.value = [imageUrl]; // 设置图片 URL 列表
  dialogVisible.value = true; // 打开对话框
}

// 关闭对话框
function closeDialog() {
  dialogVisible.value = false; // 关闭对话框
}
</script>
