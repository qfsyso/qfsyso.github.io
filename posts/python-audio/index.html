#!/usr/bin/env python3
"""
极简录音器
运行：python recorder.py
按提示输入文件名、采样率、通道数、位深、录音时长即可
"""

import pyaudio
import wave
import sys

def record_to_wav(out_file: str,
                  sample_rate: int = 16_000,
                  channels: int = 1,
                  chunk: int = 1_024,
                  record_seconds: int = 5,
                  sample_width: int = 2):
    """
    将麦克风输入保存为 WAV 文件
    :param out_file:        输出文件名，如 demo.wav
    :param sample_rate:     采样率（Hz）
    :param channels:        声道数（1=单声道 2=立体声）
    :param chunk:           每次读取的帧数
    :param record_seconds:  录音时长（秒）
    :param sample_width:    位深（字节数，2=16bit）
    """
    p = pyaudio.PyAudio()

    try:
        stream = p.open(format=p.get_format_from_width(sample_width),
                        channels=channels,
                        rate=sample_rate,
                        input=True,
                        frames_per_buffer=chunk)
        print("开始录音…（Ctrl+C 提前结束）")
        frames = []

        for _ in range(0, int(sample_rate / chunk * record_seconds)):
            data = stream.read(chunk, exception_on_overflow=False)
            frames.append(data)

        print("录音结束，正在保存…")
    except KeyboardInterrupt:
        print("\n用户中断录音，正在保存…")
    finally:
        stream.stop_stream()
        stream.close()
        p.terminate()

    with wave.open(out_file, 'wb') as wf:
        wf.setnchannels(channels)
        wf.setsampwidth(sample_width)
        wf.setframerate(sample_rate)
        wf.writeframes(b''.join(frames))
    print(f"已保存为 {out_file}")

if __name__ == "__main__":
    try:
        file_name  = input("输出文件名（*.wav）：").strip() or "output.wav"
        sr         = int(input("采样率（Hz，默认16000）：") or 16000)
        ch         = int(input("声道数（1/2，默认1）：") or 1)
        seconds    = int(input("录音时长（秒，默认5）：") or 5)
    except ValueError:
        print("输入有误，退出")
        sys.exit(1)

    record_to_wav(file_name, sample_rate=sr, channels=ch, record_seconds=seconds)

import pyaudio
import numpy as np

# 设置采集和播放参数
FORMAT = pyaudio.paInt16  # 使用16-bit PCM
CHANNELS = 1  # 单声道
RATE = 44100  # 采样率
CHUNK = 1024  # 每次读取的音频块大小（影响延迟）

# 初始化 PyAudio
p = pyaudio.PyAudio()

# 打开麦克风输入流
input_stream = p.open(format=FORMAT,
                      channels=CHANNELS,
                      rate=RATE,
                      input=True,
                      frames_per_buffer=CHUNK)

# 打开USB喇叭输出流
output_stream = p.open(format=FORMAT,
                       channels=CHANNELS,
                       rate=RATE,
                       output=True,
                       frames_per_buffer=CHUNK)

print("开始采集并播放音频...")

# 循环采集并播放音频流
try:
    while True:
        # 从麦克风采集音频数据
        audio_data = input_stream.read(CHUNK)
        
        # 播放音频到USB喇叭
        output_stream.write(audio_data)
        
except KeyboardInterrupt:
    print("\n停止采集...")

finally:
    # 停止流和关闭设备
    input_stream.stop_stream()
    input_stream.close()
    output_stream.stop_stream()
    output_stream.close()
    p.terminate()

def list_input_devices():
    print("可用输入设备列表：")
    devices = sd.query_devices()
    for idx, dev in enumerate(devices):
        if dev['max_input_channels'] > 0:
            print(f"{idx}: {dev['name']} (默认采样率: {dev['default_samplerate']} Hz)")
    print()

def record_audio(filename='output.wav', device_index=None, duration=5):
    if device_index is None:
        device_index = sd.default.device[0]  # 默认输入设备
    
    # 获取设备信息并自动采样率
    dev_info = sd.query_devices(device_index)
    samplerate = int(dev_info['default_samplerate'])

    print(f"开始录音：{duration}秒，设备：{dev_info['name']}，采样率：{samplerate} Hz")
    try:
        recording = sd.rec(int(duration * samplerate), samplerate=samplerate, channels=1, dtype='int16', device=device_index)
        sd.wait()  # 等待录音完成
        write(filename, samplerate, recording)
        print(f"录音完成，已保存为 {filename}")
    except Exception as e:
        print(f"录音失败：{e}")

if __name__ == "__main__":
    list_input_devices()
    
    try:
        device_index = int(input("请输入要使用的输入设备编号："))
        filename = input("请输入保存的文件名（例如 output.wav）：")
        duration = float(input("请输入录音时长（秒）："))
    except Exception as e:
        print("输入有误，使用默认配置")
        device_index = None
        filename = "output.wav"
        duration = 5

    record_audio(filename=filename, device_index=device_index, duration=duration)

def play_wav_on_device(filename, device_index):
    if not os.path.isfile(filename):
        print(f"文件不存在: {filename}")
        return

    try:
        samplerate, data = read(filename)
        print(f"正在使用设备 '{sd.query_devices(device_index)['name']}' 播放 {filename}")
        sd.play(data, samplerate=samplerate, device=device_index)
        sd.wait()  # 等待播放结束
        print("播放完成。")
    except Exception as e:
        print(f"播放失败: {e}")

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
talk.py  –  Debian 下麦克风→喇叭实时转发示例
python3 talk.py
"""
import sounddevice as sd
import numpy as np
import sys
import json

# ---------- 工具 ----------
def query_devices():
    """返回 sounddevice 的输入与输出设备列表"""
    return sd.query_devices()

def list_sample_rates(device_idx):
    """给定设备索引，返回它支持的采样率列表"""
    try:
        caps = sd.query_devices(device_idx)
        # 实测：对于输入设备，caps['default_samplerate'] 总是返回一个可用率
        # 为了拿到完整列表，我们尝试打开 8000~192000 之间的常用率
        rates = [8000, 11025, 16000, 22050, 32000, 44100, 48000, 88200, 96000, 192000]
        supported = []
        for r in rates:
            try:
                if caps['max_input_channels'] > 0:
                    sd.check_input_settings(device=device_idx, samplerate=r)
                else:
                    sd.check_output_settings(device=device_idx, samplerate=r)
                supported.append(r)
            except Exception:
                pass
        return supported or [int(caps['default_samplerate'])]
    except Exception as e:
        print(f"无法查询设备 {device_idx}: {e}")
        return []

def print_all_devices():
    """启动时打印所有设备及其采样率"""
    devs = query_devices()
    print("=== 发现的所有音频设备 ===")
    for idx, d in enumerate(devs):
        io = []
        if d['max_input_channels'] > 0:
            io.append("IN")
        if d['max_output_channels'] > 0:
            io.append("OUT")
        rates = list_sample_rates(idx)
        print(f"[{idx}] {d['name']}  ({','.join(io)})   支持采样率: {rates}")
    print()
def forward_stream(in_dev, out_devs, samplerate=48000, channels=1, blocksize=1024):
    """
    不依赖 CallbackQueue 的实时转发实现
    """
    import threading
    import queue

    if not out_devs:
        print("没有目标喇叭，退出")
        return

    # 共享队列：输入线程把数据放进去，每个输出线程各自取出来
    audio_q = queue.Queue()

    def input_thread():
        with sd.InputStream(device=in_dev,
                            samplerate=samplerate,
                            channels=channels,
                            dtype=np.int16,
                            blocksize=blocksize) as istream:
            while True:
                data, overflowed = istream.read(blocksize)
                if overflowed:
                    print("x", end="", flush=True)
                audio_q.put(data)

    def output_thread(dev):
        with sd.OutputStream(device=dev,
                             samplerate=samplerate,
                             channels=channels,
                             dtype=np.int16,
                             blocksize=blocksize) as ostream:
            while True:
                ostream.write(audio_q.get())

    # 启动输入线程
    t_in = threading.Thread(target=input_thread, daemon=True)
    t_in.start()

    # 为每个输出设备启动一条输出线程
    t_out = []
    for d in out_devs:
        t = threading.Thread(target=output_thread, args=(d,), daemon=True)
        t.start()
        t_out.append(t)

    print("开始实时转发，按 Ctrl+C 退出")
    try:
        t_in.join()
    except KeyboardInterrupt:
        print("\n结束")
# ---------- CLI ----------
def choose(prompt, choices):
    """简单的数字选择菜单"""
    while True:
        try:
            idx = int(input(prompt))
            if idx in choices:
                return idx
        except ValueError:
            pass
        print("无效选择，请重试")

def main():
    print_all_devices()
    devs = query_devices()
    in_devs = [i for i, d in enumerate(devs) if d['max_input_channels'] > 0]
    out_devs = [i for i, d in enumerate(devs) if d['max_output_channels'] > 0]

    if not in_devs:
        print("没有找到麦克风，退出")
        return
    if not out_devs:
        print("没有找到喇叭，退出")
        return

    print("请选择模式：")
    print("1) 把某只麦克风广播到所有喇叭")
    print("2) 把某只麦克风仅输出到指定喇叭")
    mode = choose("模式 [1/2]: ", {1, 2})

    print("请选择麦克风：")
    for i in in_devs:
        print(f"  {i}: {devs[i]['name']}")
    mic = choose("麦克风编号: ", set(in_devs))

    if mode == 1:
        targets = out_devs
    else:
        print("请选择目标喇叭：")
        for i in out_devs:
            print(f"  {i}: {devs[i]['name']}")
        targets = [choose("喇叭编号: ", set(out_devs))]

    # 默认用 48000 Hz，也可让用户输入
    sr = 48000
    print(f"使用采样率 {sr} Hz")
    forward_stream(mic, targets, samplerate=sr)

if __name__ == '__main__':
    main()

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
talk.py  –  Debian 下麦克风→喇叭实时转发示例
"""
import sounddevice as sd
import numpy as np
import argparse
import threading
import queue
import sys

# ---------- 工具 ----------
def query_devices():
    return sd.query_devices()

def list_sample_rates(device_idx):
    try:
        caps = sd.query_devices(device_idx)
        rates = [8000, 11025, 16000, 22050, 32000, 44100, 48000, 88200, 96000, 192000]
        supported = []
        for r in rates:
            try:
                if caps['max_input_channels'] > 0:
                    sd.check_input_settings(device=device_idx, samplerate=r)
                else:
                    sd.check_output_settings(device=device_idx, samplerate=r)
                supported.append(r)
            except Exception:
                pass
        return supported or [int(caps['default_samplerate'])]
    except Exception as e:
        print(f"无法查询设备 {device_idx}: {e}")
        return []

def print_all_devices():
    devs = query_devices()
    print("=== 发现的所有音频设备 ===")
    for idx, d in enumerate(devs):
        io = []
        if d['max_input_channels'] > 0:
            io.append("IN")
        if d['max_output_channels'] > 0:
            io.append("OUT")
        rates = list_sample_rates(idx)
        print(f"[{idx}] {d['name']}  ({','.join(io)})   支持采样率: {rates}")
    print()

# ---------- 实时转发 ----------
def forward_stream(in_dev, out_devs, samplerate=48000, channels=1,
                   blocksize=1024, in_gain=0.30, out_gain=0.30):
    """
    实时转发 + 音量衰减，Ctrl+C 时干净退出
    """
    if not out_devs:
        print("没有目标喇叭，退出")
        return

    audio_q   = queue.Queue(maxsize=20)   # 限长队列，防止爆音
    stop_evt  = threading.Event()         # 通知所有线程结束
    streams   = []                        # 保存所有流，便于优雅关闭

    # ---------- 输入线程 ----------
    def input_thread():
        try:
            with sd.InputStream(device=in_dev,
                                samplerate=samplerate,
                                channels=channels,
                                dtype=np.int16,
                                blocksize=blocksize) as istream:
                streams.append(istream)
                while not stop_evt.is_set():
                    data, overflowed = istream.read(blocksize)
                    if overflowed:
                        print("x", end="", flush=True)
                    data = (data * in_gain).astype(np.int16)
                    if not stop_evt.is_set():
                        audio_q.put(data, timeout=0.1)
        except queue.Full:
            pass   # 队列已满，直接丢掉一帧
        except Exception as e:
            if not stop_evt.is_set():
                print("输入线程异常:", e)

    # ---------- 输出线程 ----------
    def output_thread(dev):
        try:
            with sd.OutputStream(device=dev,
                                 samplerate=samplerate,
                                 channels=channels,
                                 dtype=np.int16,
                                 blocksize=blocksize) as ostream:
                streams.append(ostream)
                while not stop_evt.is_set():
                    try:
                        buf = audio_q.get(timeout=0.1)
                        ostream.write(buf)
                    except queue.Empty:
                        pass
        except Exception as e:
            if not stop_evt.is_set():
                print("输出线程异常:", e)

    # ---------- 启动 ----------
    threads = []

    # 输入
    t = threading.Thread(target=input_thread, daemon=True)
    t.start()
    threads.append(t)

    # 输出
    for d in out_devs:
        t = threading.Thread(target=output_thread, args=(d,), daemon=True)
        t.start()
        threads.append(t)

    print(f"开始实时转发，in_gain={in_gain}, out_gain={out_gain}，按 Ctrl+C 退出")
    try:
        for t in threads:
            t.join()
    except KeyboardInterrupt:
        print("\n exit…")
    finally:
        stop_evt.set()
        # 让阻塞在 read/write 的线程有机会跳出
        while not audio_q.empty():
            try:
                audio_q.get_nowait()
            except queue.Empty:
                break
        # 关闭所有 PortAudio 流
        for st in streams:
            try:
                st.stop()
                st.close()
            except Exception:
                pass
        print("已退出")

# ---------- CLI ----------
def choose(prompt, choices):
    while True:
        try:
            idx = int(input(prompt))
            if idx in choices:
                return idx
        except ValueError:
            pass
        print("无效选择，请重试")

    parser = argparse.ArgumentParser(description="麦克风→喇叭实时转发")
    parser.add_argument("--in-gain", type=float, default=0.30, help="输入音量系数 0~1")

    if not in_devs:
        print("没有找到麦克风，退出")
        return
    if not out_devs:
        print("没有找到喇叭，退出")
        return

    print("请选择模式：")
    print("1) 把某只麦克风广播到所有喇叭")
    print("2) 把某只麦克风仅输出到指定喇叭")
    mode = choose("模式 [1/2]: ", {1, 2})

    print("请选择麦克风：")
    for i in in_devs:
        print(f"  {i}: {devs[i]['name']}")
    mic = choose("麦克风编号: ", set(in_devs))

    if mode == 1:
        targets = out_devs
    else:
        print("请选择目标喇叭：")
        for i in out_devs:
            print(f"  {i}: {devs[i]['name']}")
        targets = [choose("喇叭编号: ", set(out_devs))]

                   in_gain=args.in_gain,
                   out_gain=args.out_gain)

    main()

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
talk.py  –  Debian 下麦克风→喇叭实时转发
用法：
  python3 talk.py                 # 默认输入/输出音量 0.3
  python3 talk.py --in-gain 0.2 --out-gain 0.15
"""
import sounddevice as sd
import numpy as np
import argparse
import threading
import queue
import sys

# ---------- 工具 ----------
def query_devices():
    return sd.query_devices()

def list_sample_rates(device_idx):

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
talk_mimo.py – 多输入多输出对话（A→B / B→A 同时）
python3 talk_mimo.py
python3 talk_mimo.py --link 0,3 --link 3,0      # 0号麦→3号喇叭, 3号麦→0号喇叭
"""
import sounddevice as sd
import numpy as np
import librosa
import argparse
import threading
import queue
import sys
import itertools

# ---------- 设备工具 ----------
def query_devices():
    return sd.query_devices()

def print_all_devices():
    devs = query_devices()
    print("=== 发现的所有音频设备 ===")
    for idx, d in enumerate(devs):
        io = []
        if d['max_input_channels'] > 0:
            io.append("IN")
        if d['max_output_channels'] > 0:
            io.append("OUT")
        print(f"[{idx}] {d['name']}  ({','.join(io)})")
    print()

# ---------- 单链路 ----------
class Link(threading.Thread):
    def __init__(self,
                 in_dev: int,
                 out_dev: int,
                 channels: int = 1,
                 blocksize: int = 1024,
                 in_gain: float = 0.30,
                 out_gain: float = 0.30):
        super().__init__(daemon=True)
        self.in_dev = in_dev
        self.out_dev = out_dev
        self.channels = channels
        self.blocksize = blocksize
        self.in_gain = in_gain
        self.out_gain = out_gain
        self._stop = threading.Event()

    def run(self):
        # 采样率
        in_sr = int(sd.query_devices(self.in_dev)['default_samplerate'])
        out_sr = int(sd.query_devices(self.out_dev)['default_samplerate'])

        q = queue.Queue(maxsize=20)

        # 输入线程
        def in_worker():
            try:
                with sd.InputStream(device=self.in_dev,
                                    samplerate=in_sr,
                                    channels=self.channels,
                                    dtype=np.int16,
                                    blocksize=self.blocksize) as istream:
                    while not self._stop.is_set():
                        data, _ = istream.read(self.blocksize)
                        data_f = data.astype(np.float32).T
                        resamp = librosa.resample(data_f,
                                                  orig_sr=in_sr,
                                                  target_sr=out_sr).T
                        resamp = (resamp * self.in_gain).astype(np.int16)
                        try:
                            q.put(resamp, timeout=0.1)
                        except queue.Full:
                            pass
            except Exception as e:
                if not self._stop.is_set():
                    print(f"输入设备 {self.in_dev} 异常: {e}")

        # 输出线程
        def out_worker():
            try:
                with sd.OutputStream(device=self.out_dev,
                                     samplerate=out_sr,
                                     channels=self.channels,
                                     dtype=np.int16,
                                     blocksize=None) as ostream:
                    while not self._stop.is_set():
                        try:
                            buf = q.get(timeout=0.1)
                            buf = (buf * self.out_gain).astype(np.int16)
                            ostream.write(buf)
                        except queue.Empty:
                            pass
            except Exception as e:
                if not self._stop.is_set():
                    print(f"输出设备 {self.out_dev} 异常: {e}")

        # 启动输入/输出两个子线程
        t_in = threading.Thread(target=in_worker, daemon=True)
        t_out = threading.Thread(target=out_worker, daemon=True)
        t_in.start(); t_out.start()
        t_in.join();  t_out.join()

    def stop(self):
        self._stop.set()

# ---------- CLI ----------
def parse_link(s):
    """解析 --link 0,3 或者 --link 1,2 这样的参数"""
    try:
        mic, spk = map(int, s.split(','))
        return mic, spk
    except Exception:
        raise argparse.ArgumentTypeError("格式应为 麦克风编号,喇叭编号")

def main():
    parser = argparse.ArgumentParser(description="多输入多输出对话")
    parser.add_argument("--link", type=parse_link, action='append',
                        help="指定一条链路 麦克风,喇叭 (可多次)")
    parser.add_argument("--in-gain", type=float, default=0.30)
    parser.add_argument("--out-gain", type=float, default=0.30)
    parser.add_argument("--channels", type=int, default=1)
    parser.add_argument("--blocksize", type=int, default=1024)
    args = parser.parse_args()

    print_all_devices()
    devs = query_devices()

    # 如果没有通过命令行指定，则进入交互式选择
    links = args.link or []
    if not links:
        print("请输入链路，格式为 麦克风编号,喇叭编号")
        print("空行结束输入")
        while True:
            line = input("链路: ").strip()
            if not line:
                break
            try:
                links.append(parse_link(line))
            except Exception as e:
                print(e)

    if not links:
        print("没有指定链路，退出")
        sys.exit(0)

    # 检查设备有效性
    for mic, spk in links:
        if mic >= len(devs) or spk >= len(devs):
            print("设备编号超出范围")
            sys.exit(1)
        if devs[mic]['max_input_channels'] == 0:
            print(f"{mic} 号设备不是麦克风")
            sys.exit(1)
        if devs[spk]['max_output_channels'] == 0:
            print(f"{spk} 号设备不是喇叭")
            sys.exit(1)

    # 启动所有链路
    threads = []
    for mic, spk in links:
        print(f"启动链路: {mic} → {spk}")
        t = Link(in_dev=mic,
                 out_dev=spk,
                 channels=args.channels,
                 blocksize=args.blocksize,
                 in_gain=args.in_gain,
                 out_gain=args.out_gain)
        t.start()
        threads.append(t)

    print("全部链路已启动，按 Ctrl+C 退出")
    try:
        for t in threads:
            t.join()
    except KeyboardInterrupt:
        print("\n正在优雅退出…")
    finally:
        for t in threads:
            t.stop()
        print("已退出")

if __name__ == '__main__':
    main()

Python Auto Restart#

Python 程序在崩溃后自动重启，确保它一直运行不间断。

方式一：在代码中实现自我重启（推荐用于简单场景）#

创建一个守护脚本 watchdog.py，用于监控主程序 your_main_program.py：

import time
import subprocess

while True:
    print("启动主程序...")
    # 启动主程序
    process = subprocess.Popen(["python", "your_main_program.py"])

    # 等待主程序结束
    process.wait()
    print("主程序崩溃或退出，5秒后重启...")
    time.sleep(5)

运行方式：

python watchdog.py

方式二：使用系统级工具管理自动重启（推荐用于正式部署）#

1. 使用 Supervisor（Linux）#

安装 Supervisor：

pip install supervisor

创建配置文件 your_supervisord.conf：

[program:your_program]
command=python your_main_program.py
autostart=true
autorestart=true
stderr_logfile=/var/log/your_program.err.log
stdout_logfile=/var/log/your_program.out.log

启动 Supervisor：

supervisord -c your_supervisord.conf

2. 使用 systemd（Linux 服务方式）#

创建 systemd 服务文件 /etc/systemd/system/your_program.service：

[Unit]
Description=My Python Program
After=network.target

[Service]
ExecStart=/usr/bin/python /path/to/your_main_program.py
Restart=always
RestartSec=5

[Install]
WantedBy=multi-user.target

启用并启动服务：

sudo systemctl daemon-reexec
sudo systemctl enable your_program
sudo systemctl start your_program

✅ 方式三：在主程序内部捕获异常避免崩溃#

也可以在主程序内部加入无限循环和异常捕获：

while True:
    try:
        # 主程序逻辑
        run_main()
    except Exception as e:
        print(f"发生异常：{e}")
        print("3秒后重启...")
        time.sleep(3)
