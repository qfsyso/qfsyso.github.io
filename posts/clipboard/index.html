共享粘贴板#

• 可以在一个设备输入文字，生成一个链接 • 其他设备在同一局域网内用链接访问可获取文字

用 Python (Flask) + 对称加密 (Fernet) 来实现

app.py#


from flask import Flask, request, redirect, url_for, render_template_string, abort
from cryptography.fernet import Fernet
import os

app = Flask(__name__)

# 生成或从环境变量读取密钥
FERNET_KEY = os.environ.get("FERNET_KEY") or Fernet.generate_key().decode()
fernet = Fernet(FERNET_KEY.encode())

# 简单内存存储 (可替换为文件或数据库)
storage = {}

# 简单页面模板
FORM_HTML = """
<!doctype html>
<title>LAN Clipboard</title>
<h1>LAN Clipboard</h1>
<form action="/share" method="post">
  <textarea name="text" rows="10" cols="50" placeholder="输入文字..."></textarea><br>
  <button type="submit">生成分享链接</button>
</form>
"""

@app.route("/", methods=["GET"])
def index():
    return FORM_HTML

@app.route("/share", methods=["POST"])
def share():
    text = request.form.get("text")
    if not text:
        return "内容不能为空", 400

    # 加密
    token = fernet.encrypt(text.encode()).decode()

    # 存储 (用 token 的 hash 做 key)
    storage[token] = text

    link = url_for("get_text", token=token, _external=True)
    return f"<p>分享链接：<a href='{link}'>{link}</a></p>"

@app.route("/get/<token>", methods=["GET"])
def get_text(token):
    try:
        # 解密验证
        text = fernet.decrypt(token.encode()).decode()
    except:
        abort(400, "无效或过期链接")
    return f"<h1>内容：</h1><pre>{text}</pre>"

if __name__ == "__main__":
    print(f"使用密钥：{FERNET_KEY} （用于加密解密，可设置为环境变量 FERNET_KEY）")
    app.run(host="0.0.0.0", port=5000)

使用#

  1. 构建镜像 在 lan-clipboard 文件夹下执行：
docker build -t lan-clipboard .
  1. 运行容器 运行并让它监听局域网：
docker run -d -p 5000:5000 --name clipboard lan-clipboard
  1. 使用方法

在浏览器访问 http://IP>:5000 （比如 http://192.168.1.10:5000） 输入文字，点击生成 复制分享的链接 在局域网其他设备用浏览器打开链接即可查看 4. 可选：自定义密钥 为了让生成的链接跨容器/多次重启可用，可以自己指定密钥：

docker run -d -p 5000:5000 -e FERNET_KEY="你的自定义密钥" lan-clipboard

🔒 加密说明 • 使用对称加密 Fernet，保证内容在 URL 中加密，不可读。 • 如果不想把内容明文存储在内存里，可只存 token，访问时解密生成内容即可。

文件 / 持久化#

• 👨‍💻 输入文字、上传文件 • 🔗 生成加密链接 • ⏰ 支持自定义过期时间（分钟） • 🗂️ SQLite 持久化，多条记录 • 🔒 文件保存到 uploads/，加密下载 app.py （美化 + 功能完整）


import os
import uuid
import hashlib
from datetime import datetime, timedelta
from flask import Flask, request, redirect, url_for, send_from_directory, abort, render_template_string
from flask_sqlalchemy import SQLAlchemy

app = Flask(__name__)

# =======================
# 配置
# =======================
BASE_DIR = os.path.abspath(os.path.dirname(__file__))

UPLOAD_FOLDER = os.path.join(BASE_DIR, 'uploads')
os.makedirs(UPLOAD_FOLDER, exist_ok=True)

DATA_FOLDER = os.path.join(BASE_DIR, 'data')
os.makedirs(DATA_FOLDER, exist_ok=True)

DATABASE_FILE = os.path.join(DATA_FOLDER, 'app.db')
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///' + DATABASE_FILE
app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False
app.config['UPLOAD_FOLDER'] = UPLOAD_FOLDER

db = SQLAlchemy(app)

# =======================
# 数据表
# =======================
class Share(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    hash = db.Column(db.String(64), unique=True, nullable=False)
    text = db.Column(db.Text, nullable=True)
    filename = db.Column(db.String(255), nullable=True)
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    expires_at = db.Column(db.DateTime, nullable=False)

# =======================
# 首页表单
# =======================
@app.route('/', methods=['GET', 'POST'])
def index():
    if request.method == 'POST':
        text = request.form.get('text')
        file = request.files.get('file')
        expires_in = int(request.form.get('expires_in', 60))  # 有效期，默认60分钟

        uid = str(uuid.uuid4()).encode('utf-8')
        hash_value = hashlib.sha256(uid).hexdigest()

        filename = None
        if file and file.filename:
            filename = f"{hash_value}_{file.filename}"
            file.save(os.path.join(UPLOAD_FOLDER, filename))

        expires_at = datetime.utcnow() + timedelta(minutes=expires_in)

        share = Share(hash=hash_value, text=text, filename=filename, expires_at=expires_at)
        db.session.add(share)
        db.session.commit()

        link = url_for('get_share', hash=hash_value, _external=True)

        return render_template_string(SUCCESS_TEMPLATE, link=link, expires_in=expires_in)

    return render_template_string(FORM_TEMPLATE)

# =======================
# 访问分享内容
# =======================
@app.route('/s/<hash>')
def get_share(hash):
    share = Share.query.filter_by(hash=hash).first()
    if not share:
        return render_template_string(ERROR_TEMPLATE, message="链接不存在")
    if datetime.utcnow() > share.expires_at:
        return render_template_string(ERROR_TEMPLATE, message="链接已过期")

    return render_template_string(SHARE_TEMPLATE, share=share)

@app.route('/uploads/<filename>')
def download_file(filename):
    return send_from_directory(UPLOAD_FOLDER, filename, as_attachment=True)

# =======================
# 页面模板
# =======================

FORM_TEMPLATE = '''
<!doctype html>
<html lang="zh-CN">
<head>
  <meta charset="utf-8">
  <title>局域网分享工具</title>
  <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/css/bootstrap.min.css" rel="stylesheet">
</head>
<body class="bg-light">
<div class="container py-5">
  <div class="row justify-content-center">
    <div class="col-lg-8">
      <div class="card shadow">
        <div class="card-body">
          <h1 class="mb-4 text-center">🔗 局域网分享工具</h1>
          <form method="post" enctype="multipart/form-data">
            <div class="mb-3">
              <label class="form-label">输入文字</label>
              <textarea name="text" class="form-control" rows="4" placeholder="请输入要分享的文字"></textarea>
            </div>
            <div class="mb-3">
              <label class="form-label">选择文件</label>
              <input type="file" name="file" class="form-control">
            </div>
            <div class="mb-3">
              <label class="form-label">过期时间（分钟）</label>
              <input type="number" name="expires_in" class="form-control" value="60" min="1">
            </div>
            <button type="submit" class="btn btn-primary w-100">生成分享链接</button>
          </form>
        </div>
      </div>
    </div>
  </div>
</div>
</body>
</html>
'''

SUCCESS_TEMPLATE = '''
<!doctype html>
<html lang="zh-CN">
<head>
  <meta charset="utf-8">
  <title>生成成功</title>
  <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/css/bootstrap.min.css" rel="stylesheet">

  <div class="mb-3">
    <input type="text" readonly class="form-control text-center" value="{{ link }}">
  </div>

  <br><br>
  <a href="/" class="btn btn-outline-secondary mt-3">返回主页</a>

'''

SHARE_TEMPLATE = '''

ERROR_TEMPLATE = '''
