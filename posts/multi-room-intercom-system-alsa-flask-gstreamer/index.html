# task_manager.py
import threading
import gi

gi.require_version('Gst', '1.0')
from gi.repository import Gst

Gst.init(None)

class TalkSession(threading.Thread):
    def __init__(self, input_dev, output_dev):
        super().__init__()
        self.input_dev = input_dev
        self.output_dev = output_dev
        self.pipeline = None
        self.running = False

    def run(self):
        self.pipeline = Gst.parse_launch(
            f"alsasrc device={self.input_dev} ! audioconvert ! audioresample ! alsasink device={self.output_dev}"
        )
        self.pipeline.set_state(Gst.State.PLAYING)
        self.running = True
        bus = self.pipeline.get_bus()
        while self.running:
            msg = bus.timed_pop_filtered(100 * Gst.MSECOND, Gst.MessageType.ERROR | Gst.MessageType.EOS)
            if msg:
                break

    def stop(self):
        self.running = False
        if self.pipeline:
            self.pipeline.set_state(Gst.State.NULL)

class TalkManager:
    def __init__(self):
        self.sessions = {}

    def start_talk(self, from_dev, to_dev):
        key = f"{from_dev}->{to_dev}"
        if key not in self.sessions:
            session = TalkSession(from_dev, to_dev)
            self.sessions[key] = session
            session.start()
            return True
        return False

    def stop_talk(self, from_dev, to_dev):
        key = f"{from_dev}->{to_dev}"
        if key in self.sessions:
            self.sessions[key].stop()
            self.sessions[key].join()
            del self.sessions[key]
            return True
        return False

    def list_sessions(self):
        return list(self.sessions.keys())

from flask import Flask, jsonify, request, send_from_directory, render_template

from talk_manager import TalkManager

app = Flask(__name__, static_folder='templates', static_url_path='')
manager = TalkManager()

def get_audio_devices():
    result = subprocess.run(['arecord', '-l'], stdout=subprocess.PIPE)
    inputs = []
    for line in result.stdout.decode().split('\n'):
        if 'card' in line:
            parts = line.strip().split()
            card = parts[1].replace(':', '')
            device = parts[3].replace(':', '')
            hw_name = f'hw:{card},{device}'
            inputs.append(hw_name)
    return inputs

@app.route('/')
def index():
    return send_from_directory('templates', 'index.html')

@app.route('/api/devices')
def list_devices():
    devices = get_audio_devices()
    return jsonify({'devices': devices})

@app.route('/api/start', methods=['POST'])
def start_talk():
    data = request.json
    from_dev = data['from']
    to_dev = data['to']
    success = manager.start_talk(from_dev, to_dev)
    return jsonify({'success': success})

@app.route('/api/stop', methods=['POST'])
def stop_talk():
    data = request.json
    from_dev = data['from']
    to_dev = data['to']
    success = manager.stop_talk(from_dev, to_dev)
    return jsonify({'success': success})

@app.route('/api/sessions')
def list_sessions():
    return jsonify({'sessions': manager.list_sessions()})

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)

# talk_manager.py
import subprocess
import re
import gi
import threading
gi.require_version('Gst', '1.0')
from gi.repository import Gst, GLib

Gst.init(None)

def list_alsa_devices(device_type='playback'):
    cmd = ['aplay', '-l'] if device_type == 'playback' else ['arecord', '-l']
    try:
        result = subprocess.run(cmd, capture_output=True, text=True, check=True)
        output = result.stdout
    except Exception as e:
        print(f"Error listing ALSA devices: {e}")
        return []

    devices = []
    regex = re.compile(r"card (\d+): ([^\[]+)\[.*?\], device (\d+): ([^\[]+)\[.*?\]")
    for line in output.splitlines():
        m = regex.search(line)
        if m:
            card = int(m.group(1))
            card_name = m.group(2).strip()
            device = int(m.group(3))
            device_name = m.group(4).strip()
            devices.append((card, device, f"{card_name} {device_name}"))
    return devices

class TalkManager(threading.Thread):
    def __init__(self, input_dev=None, output_dev=None, broadcast_dev=None,
                 input_volume=1.0, output_volume=1.0, broadcast_volume=1.0):
        super().__init__()
        self.loop = GLib.MainLoop()
        self.pipeline = None
        self.running = False

        self.input_devices = list_alsa_devices('capture')
        self.output_devices = list_alsa_devices('playback')

        if not self.input_devices or not self.output_devices:
            raise RuntimeError("找不到录音或播放设备")

        self.input_device = input_dev or self.input_devices[0]
        self.output_device = output_dev or self.output_devices[0]
        self.broadcast_device = broadcast_dev or self.output_devices[0]

        self.input_volume = input_volume
        self.output_volume = output_volume
        self.broadcast_volume = broadcast_volume

    def run(self):
        self.running = True

        in_dev_str = f"hw:{self.input_device[0]},{self.input_device[1]}"
        out_dev_str = f"hw:{self.output_device[0]},{self.output_device[1]}"
        bc_dev_str = f"hw:{self.broadcast_device[0]},{self.broadcast_device[1]}"

        pipeline_str = (
            f"alsasrc device={in_dev_str} ! volume volume={self.input_volume} ! audioconvert ! audioresample ! tee name=t "
            f"t. ! queue ! volume volume={self.output_volume} ! audioconvert ! audioresample ! alsasink device={out_dev_str} "
            f"t. ! queue ! volume volume={self.broadcast_volume} ! audioconvert ! audioresample ! alsasink device={bc_dev_str}"
        )
        print(f"启动管道: {pipeline_str}")

        self.pipeline = Gst.parse_launch(pipeline_str)
        self.pipeline.set_state(Gst.State.PLAYING)

        try:
            self.loop.run()
        except Exception as e:
            print("GStreamer loop exited:", e)
        finally:
            self.pipeline.set_state(Gst.State.NULL)
            self.running = False

    def stop(self):
        if self.running:
            self.loop.quit()
            self.join()
            self.running = False

from flask import Flask, jsonify, request
from talk_manager import TalkManager, list_alsa_devices
import threading

app = Flask(__name__, static_folder='frontend', static_url_path='/')

talk_manager = None
tm_lock = threading.Lock()

@app.route('/')
def index():
    return app.send_static_file('index.html')

@app.route('/api/devices')
def api_devices():
    inputs = list_alsa_devices('capture')
    outputs = list_alsa_devices('playback')
    return jsonify({
        'inputs': [{'card': c, 'device': d, 'name': n} for c,d,n in inputs],
        'outputs': [{'card': c, 'device': d, 'name': n} for c,d,n in outputs],
    })

@app.route('/api/start', methods=['POST'])
def api_start():
    global talk_manager
    data = request.json
    input_dev = data.get('input')  # {"card":1,"device":0}
    output_dev = data.get('output')
    broadcast_dev = data.get('broadcast')
    input_volume = float(data.get('input_volume', 1.0))
    output_volume = float(data.get('output_volume', 1.0))
    broadcast_volume = float(data.get('broadcast_volume', 1.0))

    with tm_lock:
        if talk_manager and talk_manager.running:
            return jsonify({'status': 'already running'})

        in_dev = (input_dev['card'], input_dev['device']) if input_dev else None
        out_dev = (output_dev['card'], output_dev['device']) if output_dev else None
        bc_dev = (broadcast_dev['card'], broadcast_dev['device']) if broadcast_dev else None

        talk_manager = TalkManager(
            input_dev=in_dev,
            output_dev=out_dev,
            broadcast_dev=bc_dev,
            input_volume=input_volume,
            output_volume=output_volume,
            broadcast_volume=broadcast_volume
        )
        talk_manager.start()
    return jsonify({'status': 'started'})

@app.route('/api/stop', methods=['POST'])
def api_stop():
    global talk_manager
    with tm_lock:
        if talk_manager and talk_manager.running:
            talk_manager.stop()
            talk_manager = None
            return jsonify({'status': 'stopped'})
        else:
            return jsonify({'status': 'not running'})

@app.route('/api/status')
def api_status():
    global talk_manager
    running = talk_manager.running if talk_manager else False
    return jsonify({'running': running})

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)

from flask import Flask, jsonify, request
import subprocess
import re
import threading
import gi

gi.require_version('Gst', '1.0')
from gi.repository import Gst, GLib

Gst.init(None)

app = Flask(__name__, static_folder='frontend', static_url_path='/')

def list_alsa_devices(device_type='playback'):
    cmd = ['aplay', '-l'] if device_type == 'playback' else ['arecord', '-l']
    try:
        result = subprocess.run(cmd, capture_output=True, text=True, check=True)
        output = result.stdout
    except Exception as e:
        print(f"Error listing ALSA devices: {e}")
        return []

    devices = []
    regex = re.compile(r"card (\d+): ([^\[]+)\[.*?\], device (\d+): ([^\[]+)\[.*?\]")
    for line in output.splitlines():
        m = regex.search(line)
        if m:
            card = int(m.group(1))
            card_name = m.group(2).strip()
            device = int(m.group(3))
            device_name = m.group(4).strip()
            devices.append({
                "card": card,
                "device": device,
                "name": f"{card_name} {device_name}"
            })
    return devices

class TalkManager(threading.Thread):
    def __init__(self, input_dev, output_dev, broadcast_devs, input_volume=1.0, output_volume=1.0, broadcast_volume=1.0):
        super().__init__()
        self.loop = GLib.MainLoop()
        self.pipeline = None
        self.running = False

        self.input_device = input_dev
        self.output_device = output_dev
        self.broadcast_devices = broadcast_devs or []

        self.input_volume = input_volume
        self.output_volume = output_volume
        self.broadcast_volume = broadcast_volume

    def run(self):
        self.running = True

        in_dev_str = f"hw:{self.input_device['card']},{self.input_device['device']}"
        out_dev_str = f"hw:{self.output_device['card']},{self.output_device['device']}"

        pipeline_str = (
            f"alsasrc device={in_dev_str} ! volume volume={self.input_volume} ! audioconvert ! audioresample ! tee name=t "
            f"t. ! queue ! volume volume={self.output_volume} ! audioconvert ! audioresample ! alsasink device={out_dev_str} "
        )

        for bc_dev in self.broadcast_devices:
            bc_dev_str = f"hw:{bc_dev['card']},{bc_dev['device']}"
            pipeline_str += f" t. ! queue ! volume volume={self.broadcast_volume} ! audioconvert ! audioresample ! alsasink device={bc_dev_str} "

        print(f"启动管道: {pipeline_str}")

        self.pipeline = Gst.parse_launch(pipeline_str)
        self.pipeline.set_state(Gst.State.PLAYING)

        try:
            self.loop.run()
        except Exception as e:
            print(f"GStreamer loop exited: {e}")
        finally:
            self.pipeline.set_state(Gst.State.NULL)
            self.running = False

    def stop(self):
        if self.running:
            self.loop.quit()
            self.join()
            self.running = False

talk_managers = {}  # 多个通道
tm_lock = threading.Lock()
next_id = 1

@app.route('/')
def index():
    return app.send_static_file('index.html')

@app.route('/api/devices')
def api_devices():
    inputs = list_alsa_devices('capture')
    outputs = list_alsa_devices('playback')
    return jsonify({'inputs': inputs, 'outputs': outputs})

@app.route('/api/start', methods=['POST'])
def api_start():
    global next_id
    data = request.json
    input_dev = data.get('input')
    output_dev = data.get('output')
    broadcast_devs = data.get('broadcasts', [])

    input_volume = float(data.get('input_volume', 1.0))
    output_volume = float(data.get('output_volume', 1.0))
    broadcast_volume = float(data.get('broadcast_volume', 1.0))

    if not input_dev or not output_dev:
        return jsonify({'status': 'error', 'message': '输入或输出设备未选择'})

    tm = TalkManager(
        input_dev=input_dev,
        output_dev=output_dev,
        broadcast_devs=broadcast_devs,
        input_volume=input_volume,
        output_volume=output_volume,
        broadcast_volume=broadcast_volume
    )

    with tm_lock:
        tm_id = next_id
        next_id += 1
        talk_managers[tm_id] = tm
        tm.start()

    return jsonify({'status': 'started', 'id': tm_id})

@app.route('/api/stop', methods=['POST'])
def api_stop():
    data = request.json
    tm_id = data.get('id')

    with tm_lock:
        tm = talk_managers.get(tm_id)
        if tm and tm.running:
            tm.stop()
            del talk_managers[tm_id]
            return jsonify({'status': 'stopped'})
        else:
            return jsonify({'status': 'not running'})

@app.route('/api/status')
def api_status():
    with tm_lock:
        active = []
        for tm_id, tm in talk_managers.items():
            if tm.running:
                active.append({
                    'id': tm_id,
                    'input': tm.input_device,
                    'output': tm.output_device
                })
    return jsonify({'active': active})

    app.run(host='0.0.0.0', port=5000)
