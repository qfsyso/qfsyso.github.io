WAV STT#

Python Flask 框架

SQLite 做存储

支持批量上传音频文件

后端调用开源 STT 库

支持音频回放和文本搜索


/app
├── app.py # Flask 后端
├── templates/
│ └── index.html # 前端页面
├── static/
│ └── audio/ # 上传音频存储目录（Docker 内）
├── requirements.txt # 依赖
├── Dockerfile
└── stt_utils.py # STT 相关工具函数

后端核心功能#

/：首页，返回上传页面

/upload：接受多文件上传，保存音频，调用 STT 转文本，保存结果进 SQLite

/audio/：提供音频文件访问，前端可回放

/search：文本搜索接口，返回匹配记录

SQLite 数据库保存：字段 id, filename, upload_time, transcript

app.py


import os
import sqlite3
from datetime import datetime
from flask import Flask, request, render_template, send_from_directory, jsonify
import whisper

app = Flask(**name**)
app.config['UPLOAD_FOLDER'] = 'static/audio'
os.makedirs(app.config['UPLOAD_FOLDER'], exist_ok=True)

DB = 'database.db'

# 初始化模型（加载一次）

model = whisper.load_model("base")

def init_db():
with sqlite3.connect(DB) as conn:
conn.execute('''
CREATE TABLE IF NOT EXISTS recordings (
id INTEGER PRIMARY KEY AUTOINCREMENT,
filename TEXT,
upload_time TEXT,
transcript TEXT
)
''')

def insert_record(filename, transcript):
with sqlite3.connect(DB) as conn:
conn.execute("INSERT INTO recordings (filename, upload_time, transcript) VALUES (?, ?, ?)",
(filename, datetime.now().isoformat(), transcript))
conn.commit()

def search_transcripts(keyword):
with sqlite3.connect(DB) as conn:
cursor = conn.execute("SELECT id, filename, upload_time, transcript FROM recordings WHERE transcript LIKE ?", ('%'+keyword+'%',))
return cursor.fetchall()

def transcribe_audio(path): # whisper 转文本
result = model.transcribe(path)
return result["text"]

@app.route('/')
def index():
return render_template('index.html')

@app.route('/upload', methods=['POST'])
def upload():
files = request.files.getlist('files')
results = []
for file in files:
filename = file.filename
save_path = os.path.join(app.config['UPLOAD_FOLDER'], filename)
file.save(save_path)
transcript = transcribe_audio(save_path)
insert_record(filename, transcript)
results.append({'filename': filename, 'transcript': transcript})
return jsonify(results)

@app.route('/audio/<filename>')
def audio(filename):
return send_from_directory(app.config['UPLOAD_FOLDER'], filename)

@app.route('/search')
def search():
keyword = request.args.get('q', '')
if not keyword:
return jsonify([])
records = search_transcripts(keyword)
results = [{'id': r[0], 'filename': r[1], 'upload_time': r[2], 'transcript': r[3]} for r in records]
return jsonify(results)

if **name** == '**main**':
init_db()
app.run(host='0.0.0.0', port=5000)

model = whisper.load_model("base")

def init_db():
with sqlite3.connect(DB) as conn:
conn.execute('''
CREATE TABLE IF NOT EXISTS recordings (
id INTEGER PRIMARY KEY AUTOINCREMENT,
filename TEXT,
upload_time TEXT,
transcript TEXT
)
''')

def insert_record(filename, transcript):
with sqlite3.connect(DB) as conn:
conn.execute(
"INSERT INTO recordings (filename, upload_time, transcript) VALUES (?, ?, ?)",
(filename, datetime.now().isoformat(), transcript)
)
conn.commit()

def search*transcripts(keyword, page=1, per_page=5):
offset = (page - 1) * per*page
with sqlite3.connect(DB) as conn:
if keyword:
cursor = conn.execute(
"SELECT COUNT(*) FROM recordings WHERE transcript LIKE ?",
('%' + keyword + '%',)
)
total = cursor.fetchone()[0]

            cursor = conn.execute(
                "SELECT id, filename, upload_time, transcript FROM recordings WHERE transcript LIKE ? ORDER BY id DESC LIMIT ? OFFSET ?",
                ('%' + keyword + '%', per_page, offset)
            )
        else:
            cursor = conn.execute("SELECT COUNT(*) FROM recordings")
            total = cursor.fetchone()[0]

            cursor = conn.execute(
                "SELECT id, filename, upload_time, transcript FROM recordings ORDER BY id DESC LIMIT ? OFFSET ?",
                (per_page, offset)
            )
        results = cursor.fetchall()
    return results, total

def transcribe_audio(path):
result = model.transcribe(path)
return result["text"]

@app.route('/')

return render_template('index.html')

page = int(request.args.get('page', 1))
per_page = 5

    records, total = search_transcripts(keyword, page, per_page)
    total_pages = (total + per_page - 1) // per_page

            <div class="transcript">${item.transcript}</div>

            pagination.innerHTML += `<button onclick="changePage(${
              page + 1

          }

        search(p);

        search(1);

扩展：分离模型#

提前下载模型文件，用 local_files_only=True 加载 可以在有外网的地方下载模型，然后打包复制到目标机器使用。

步骤如下：

在有外网的机器：#


from speechbrain.inference.separation import SepformerSeparation as separator
# 第一次运行，会自动下载模型
model = separator.from_hparams(source="speechbrain/sepformer-wsj02mix", savedir="pretrained_models/sepformer-wsj02mix")

print("模型下载完毕")

下载后的模型会存储在：#

Linux: ~/.cache/huggingface/hub 或指定的 savedir="…" 路径中

把该文件夹复制到目标服务器#

/root/room/pretrained_models/sepformer-wsj02mix

修改代码为离线加载：#


from speechbrain.inference.separation import SepformerSeparation as separator

# 离线加载模型（不会尝试联网）
model = separator.from_hparams(
    source="/root/room/pretrained_models/sepformer-wsj02mix",
    savedir="/root/room/pretrained_models/sepformer-wsj02mix",
    run_opts={"local_files_only": True}
)

# 分离的语音文件
est_sources = model.separate_file(path="多人混音.wav")

...

注意#

  1. Sepformer 模型输入必须是 8kHz 采样率 SpeechBrain 官方预训练的分离模型（如 sepformer-wsj02mix）只接受 8kHz 的输入，否则会出错或效果异常。

所以我们必须在模型前 将音频从 48kHz 降采样到 8kHz，供模型处理。

  1. 模型输出的音频也是 8kHz 所以在 torchaudio.save() 保存输出的时候默认是 8kHz。

print("Transcription:", text)

不同模型的额外示例 也可以尝试其他语言或架构的模型，例如：

意大利语模型（基于 wav2vec2 + CTC/Attention）：#


from speechbrain.inference.ASR import EncoderDecoderASR
asr_model = EncoderDecoderASR.from_hparams(
source="speechbrain/asr-wav2vec2-commonvoice-it",
savedir="pretrained_models/asr-wav2vec2-commonvoice-it"
)
text = asr_model.transcribe_file("example-it.wav")
print(text)

普通话（AISHELL）模型：#


from speechbrain.inference.interfaces import foreign_class
asr_model = foreign_class(
source="speechbrain/asr-wav2vec2-ctc-aishell",
pymodule_file="custom_interface.py",
classname="CustomEncoderDecoderASR"
)
text = asr_model.transcribe_file("example.wav")
print(text)
