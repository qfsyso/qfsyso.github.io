.NET 单例缓存

模拟数据库用户信息查询#

using Microsoft.Extensions.Caching.Memory;
using Microsoft.Extensions.Logging;
using System;

public static class CacheManager
{
//    // 缓存实例
    // private static readonly IMemoryCache _cache = new MemoryCache(new MemoryCacheOptions
    // {
    //     SizeLimit = 1024 // 限制缓存最大条目数
    // });
    private static readonly MemoryCache _cache = new MemoryCache(new MemoryCacheOptions());

    private static readonly ILogger _logger = LoggerFactory.Create(builder => 
        builder.AddConsole()).CreateLogger("CacheManager");

    /// <summary>
    /// 获取或创建缓存项（泛型方法）
    /// </summary>
    public static T GetOrCreate<T>(string key, Func<ICacheEntry, T> factory, TimeSpan? expiration = null)
    {
        try
        {
            return _cache.GetOrCreate(key, entry =>
            {
                // 设置缓存策略
                entry.SetSize(1); // 每个条目占1单位空间
                if (expiration != null)
                {
                    entry.AbsoluteExpirationRelativeToNow = expiration;
                }
                else
                {
                    // 默认缓存策略：5分钟绝对过期 + 2分钟滑动过期
                    entry.AbsoluteExpirationRelativeToNow = TimeSpan.FromMinutes(5);
                    entry.SlidingExpiration = TimeSpan.FromMinutes(2);
                }

                _logger.LogInformation($"缓存未命中，正在生成数据: {key}");
                return factory(entry);
            });
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"缓存操作失败: {key}");
            throw;
        }
    }

    /// <summary>
    /// 清除指定缓存项
    /// </summary>
    public static void Remove(string key) => _cache.Remove(key);

    /// <summary>
    /// 清空所有缓存
    /// </summary>
    public static void Clear() => _cache.Compact(1.0); // 释放100%缓存
}

/// <summary>
/// 模拟实际业务逻辑：用户信息服务
/// </summary>
public static class UserService
{
    /// <summary>
    /// 获取用户信息（带缓存）
    /// </summary>
    public static UserInfo GetUserInfo(int userId)
    {
        return CacheManager.GetOrCreate(
            key: $"UserInfo_{userId}",
            factory: entry =>
            {
                // 模拟实际业务逻辑：数据库查询
                var user = QueryDatabase(userId);
                
                // 设置缓存依赖：如果用户被修改则立即过期
                entry.AddExpirationToken(new UserChangeToken(userId));
                return user;
            },
            expiration: TimeSpan.FromMinutes(10) // 特定业务缓存时间
        );
    }

    // 模拟数据库查询
    private static UserInfo QueryDatabase(int userId)
    {
        // 模拟耗时操作
        System.Threading.Thread.Sleep(1000);
        
        return new UserInfo(
            Id: userId,
            Name: $"User_{userId}",
            LastLogin: DateTime.UtcNow,
            Email: $"{userId}@example.com"
        );
    }
}

/// <summary>
/// 用户信息模型
/// </summary>
public record UserInfo(
    int Id,
    string Name,
    DateTime LastLogin,
    string Email
);

/// <summary>
/// 自定义缓存依赖：当用户数据变更时触发缓存过期
/// </summary>
public class UserChangeToken : Microsoft.Extensions.Primitives.IChangeToken
{
    private readonly int _userId;
    private bool _hasChanged;

    public UserChangeToken(int userId) => _userId = userId;

    public bool HasChanged => _hasChanged;
    public bool ActiveChangeCallbacks => false;

    // 实际项目中可调用此方法触发过期
    public void NotifyChange() => _hasChanged = true;

    public IDisposable RegisterChangeCallback(Action<object> callback, object state) 
        => Microsoft.Extensions.Primitives.EmptyDisposable.Instance;
}

使用示例#

var user = UserService.GetUserInfo(123);


// 第一次调用会查询数据库并缓存
var user1 = UserService.GetUserInfo(123);

// 第二次调用直接返回缓存
var user2 = UserService.GetUserInfo(123);

// 当用户数据更新时（如修改邮箱）
UpdateUserInDatabase(123);
new UserChangeToken(123).NotifyChange(); // 手动触发缓存失效

// 下次调用将重新查询
var user3 = UserService.GetUserInfo(123);
