四、popup.js（所有交互逻辑）#


/* global chrome */
const $ = (sel, ctx = document) => ctx.querySelector(sel);

/* ---------- 存储工具 ---------- */
const STORAGE_KEY = ' L-ToDo';
async function save(data) {
  return chrome.storage.local.set({ [STORAGE_KEY]: data });
}
async function load() {
  const res = await chrome.storage.local.get(STORAGE_KEY);
  return res[STORAGE_KEY] || { links: [], todos: [] };
}

/* ---------- 初始化 ---------- */
(async () => {
  const data = await load();
  renderLinks(data.links);
  renderTodos(data.todos);
  greet();
})();

/* ---------- 问候 ---------- */
function greet() {
  const name = localStorage.getItem('username') || 'friend';
  $('#greeting').textContent = `Hello, ${name}!`;
}

/* ---------- Quick Links ---------- */
$('#linkForm').addEventListener('submit', async e => {
  e.preventDefault();
  const name = $('#linkName').value.trim();
  const url  = $('#linkUrl').value.trim();
  const data = await load();
  data.links.push({ name, url });
  await save(data);
  renderLinks(data.links);
  e.target.reset();
});

function renderLinks(list) {
  const ul = $('#linkList');
  ul.innerHTML = '';
  list.forEach(({ name, url }, idx) => {
    const li = document.createElement('li');
    li.innerHTML = `
      <a href="${url}" target="_blank" rel="noopener">${name}</a>
      <button data-link-idx="${idx}">✕</button>`;
    ul.appendChild(li);
  });
  ul.addEventListener('click', async e => {
    if (!e.target.matches('[data-link-idx]')) return;
    const idx = +e.target.dataset.linkIdx;
    const data = await load();
    data.links.splice(idx, 1);
    await save(data);
    renderLinks(data.links);
  });
}

/* ---------- To-Do List ---------- */
$('#todoForm').addEventListener('submit', async e => {
  e.preventDefault();
  const text = $('#todoInput').value.trim();
  const data = await load();
  data.todos.push({ text, done: false });
  await save(data);
  renderTodos(data.todos);
  e.target.reset();
});

function renderTodos(list) {
  const ul = $('#todoList');
  ul.innerHTML = '';
  list.forEach(({ text, done }, idx) => {
    const li = document.createElement('li');
    li.innerHTML = `
      <label>
        <input type="checkbox" data-todo-idx="${idx}" ${done ? 'checked' : ''}>
        <span style="${done ? 'text-decoration:line-through' : ''}">${text}</span>
      </label>
      <button data-todo-del="${idx}">✕</button>`;
    ul.appendChild(li);
  });
  ul.addEventListener('click', async e => {
    const data = await load();
    if (e.target.matches('[data-todo-idx]')) {
      const idx = +e.target.dataset.todoIdx;
      data.todos[idx].done = e.target.checked;
      await save(data);
      renderTodos(data.todos);
    }
    if (e.target.matches('[data-todo-del]')) {
      const idx = +e.target.dataset.todoDel;
      data.todos.splice(idx, 1);
      await save(data);
      renderTodos(data.todos);
    }
  });
}

/* ---------- 导出 / 导入 ---------- */
$('#exportBtn').addEventListener('click', async () => {
  const blob = new Blob([JSON.stringify(await load(), null, 2)], { type: 'application/json' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = ' L-ToDo-backup.json';
  a.click();
  URL.revokeObjectURL(url);
});

$('#importBtn').addEventListener('click', () => $('#importFile').click());
$('#importFile').addEventListener('change', async e => {
  const file = e.target.files[0];
  if (!file) return;
  try {
    const data = JSON.parse(await file.text());
    await save(data);
    location.reload();
  } catch (err) {
    alert('Invalid backup file');
  }
});

安装步骤#

将以上文件按目录放好。 打开 Chrome → 右上角「┇」→ 更多工具 → 扩展程序 → 右上角开启「开发者模式」。 点击「加载已解压的扩展程序」，选择 L-ToDo 文件夹。 新建标签页即可看到效果。 至此，一个支持「自定义快速链接 + To-Do 列表 + 配置导出/导入」的极简 new-tab 插件就完成了。

popup.js（在原有基础上追加实时时间）#


/* … 上方保持原样 … */

/* ---------- 实时时钟 ---------- */
function updateClock() {
  const now = new Date();
  const hh = String(now.getHours()).padStart(2, '0');
  const mm = String(now.getMinutes()).padStart(2, '0');
  const ss = String(now.getSeconds()).padStart(2, '0');
  $('#clock').textContent = `${hh}:${mm}:${ss}`;
}
updateClock();
setInterval(updateClock, 1000);
