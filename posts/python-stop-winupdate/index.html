import winreg
import subprocess

REG_PATH = r"SOFTWARE\Policies\Microsoft\Windows\WindowsUpdate\AU"
REG_NAME = "NoAutoUpdate"

def get_update_status():
    """获取 Windows 自动更新状态"""
    try:
        reg_key = winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE, REG_PATH, 0, winreg.KEY_READ)
        value, _ = winreg.QueryValueEx(reg_key, REG_NAME)
        winreg.CloseKey(reg_key)

        if value == 1:
            return "Windows 自动更新已禁用 (注册表控制)"
        else:
            return "Windows 自动更新已启用 (注册表控制)"
    except FileNotFoundError:
        return "未设置策略，使用系统默认 (通常为启用自动更新)"


def disable_windows_update():
    """禁用 Windows 自动更新"""
    try:
        reg_key = winreg.CreateKey(winreg.HKEY_LOCAL_MACHINE, REG_PATH)
        winreg.SetValueEx(reg_key, REG_NAME, 0, winreg.REG_DWORD, 1)
        winreg.CloseKey(reg_key)
        refresh_group_policy()
        return "已禁用 Windows 自动更新 (注册表方式)"
    except Exception as e:
        return f"禁用失败: {e}"


def enable_windows_update():
    """启用 Windows 自动更新"""
    try:
        reg_key = winreg.CreateKey(winreg.HKEY_LOCAL_MACHINE, REG_PATH)
        winreg.SetValueEx(reg_key, REG_NAME, 0, winreg.REG_DWORD, 0)
        winreg.CloseKey(reg_key)
        refresh_group_policy()
        return "已启用 Windows 自动更新 (注册表方式)"
    except Exception as e:
        return f"启用失败: {e}"


def refresh_group_policy():
    """刷新组策略"""
    try:
        print("正在刷新组策略，请稍候...")
        subprocess.run(["gpupdate", "/force"], capture_output=True, text=True, check=True)

    except Exception as e:
        print(f"刷新组策略失败: {e}")

import winreg
import subprocess
from datetime import datetime, timedelta

REG_PATH = r"SOFTWARE\Policies\Microsoft\Windows\WindowsUpdate\AU"

def get_update_status():
    """获取 Windows 更新策略状态"""
    try:
        reg_key = winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE, REG_PATH, 0, winreg.KEY_READ)
        try:
            no_auto, _ = winreg.QueryValueEx(reg_key, "NoAutoUpdate")
        except FileNotFoundError:
            no_auto = 0
        try:
            pause_start, _ = winreg.QueryValueEx(reg_key, "PauseUpdatesStartTime")
            pause_end, _ = winreg.QueryValueEx(reg_key, "PauseUpdatesEndTime")
        except FileNotFoundError:
            pause_start, pause_end = None, None
        winreg.CloseKey(reg_key)

        status = []
        if no_auto == 1:
            status.append("自动更新已禁用")
        else:
            status.append("自动更新已启用")

        if pause_start and pause_end:
            status.append(f"更新已暂停: {pause_start} → {pause_end}")
        return "；".join(status)
    except Exception:
        return "未设置策略，系统默认 (自动更新启用)"


def disable_windows_update():
    set_reg_value("NoAutoUpdate", 1, winreg.REG_DWORD)
    refresh_group_policy()
    return "已禁用 Windows 自动更新"


def enable_windows_update():
    set_reg_value("NoAutoUpdate", 0, winreg.REG_DWORD)
    refresh_group_policy()
    return "已启用 Windows 自动更新"


def pause_updates(days=7):
    """暂停更新 N 天"""
    start_date = datetime.today().strftime("%Y-%m-%d")
    end_date = (datetime.today() + timedelta(days=days)).strftime("%Y-%m-%d")
    set_reg_value("PauseUpdatesStartTime", start_date, winreg.REG_SZ)
    set_reg_value("PauseUpdatesEndTime", end_date, winreg.REG_SZ)
    refresh_group_policy()
    return f"已暂停 Windows 更新 {days} 天 ({start_date} → {end_date})"


def resume_updates():
    """恢复更新 (删除暂停设置)"""
    delete_reg_value("PauseUpdatesStartTime")
    delete_reg_value("PauseUpdatesEndTime")
    refresh_group_policy()
    return "已恢复 Windows 更新 (取消暂停)"


def query_pause_remaining():
    """查询暂停剩余天数"""
    try:
        reg_key = winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE, REG_PATH, 0, winreg.KEY_READ)
        pause_end, _ = winreg.QueryValueEx(reg_key, "PauseUpdatesEndTime")
        winreg.CloseKey(reg_key)

        end_date = datetime.strptime(pause_end, "%Y-%m-%d")
        today = datetime.today()
        delta = (end_date - today).days

        if delta > 0:
            return f"更新暂停还剩 {delta} 天 (到 {pause_end})"
        elif delta == 0:
            return f"更新暂停今天到期 ({pause_end})"
        else:
            return "暂停已过期"
    except FileNotFoundError:
        return "当前未设置暂停更新"
    except Exception as e:
        return f"查询失败: {e}"


def set_reg_value(name, value, reg_type):
    try:
        reg_key = winreg.CreateKey(winreg.HKEY_LOCAL_MACHINE, REG_PATH)
        winreg.SetValueEx(reg_key, name, 0, reg_type, value)
        winreg.CloseKey(reg_key)
    except Exception as e:
        print(f"写入注册表失败: {e}")


def delete_reg_value(name):
    try:
        reg_key = winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE, REG_PATH, 0, winreg.KEY_SET_VALUE)
        winreg.DeleteValue(reg_key, name)
        winreg.CloseKey(reg_key)
    except FileNotFoundError:
        pass


def refresh_group_policy():
    """刷新组策略"""
    try:
        print("正在刷新组策略...")
        subprocess.run(["gpupdate", "/force"], capture_output=True, text=True, check=True)
        print("组策略刷新完成 ✅")
    except Exception as e:
        print(f"刷新组策略失败: {e}")


if __name__ == "__main__":
    print("请选择操作：")
    print("1 - 查看 Windows 更新状态")
    print("2 - 禁用自动更新")
    print("3 - 启用自动更新")
    print("4 - 暂停更新 (延迟 N 天)")
    print("5 - 恢复更新 (取消暂停)")
    print("6 - 查询暂停剩余天数")

    choice = input("请输入选项 (1/2/3/4/5/6): ").strip()

    if choice == "1":
        print(get_update_status())
    elif choice == "2":
        print(disable_windows_update())
    elif choice == "3":
        print(enable_windows_update())
    elif choice == "4":
        days = input("请输入要暂停的天数 (默认 7): ").strip()
        days = int(days) if days.isdigit() else 7
        print(pause_updates(days))
    elif choice == "5":
        print(resume_updates())
    elif choice == "6":
        print(query_pause_remaining())
    else:
        print("无效选项")

启动项#

import os
import winreg
import shutil

def get_startup_folder(user=True):
    """获取启动文件夹路径"""
    if user:
        return os.path.expandvars(r"%APPDATA%\Microsoft\Windows\Start Menu\Programs\Startup")
    else:
        return os.path.expandvars(r"%ProgramData%\Microsoft\Windows\Start Menu\Programs\Startup")

def list_startup_items():
    """列出所有启动项（文件夹 + 注册表）"""
    items = []

    # 启动文件夹
    for user in [True, False]:
        folder = get_startup_folder(user)
        if os.path.exists(folder):
            for f in os.listdir(folder):
                items.append({
                    "type": "folder",
                    "user": "CurrentUser" if user else "AllUsers",
                    "name": f,
                    "path": os.path.join(folder, f)
                })

    # 注册表启动项
    reg_paths = [
        (winreg.HKEY_CURRENT_USER, r"Software\Microsoft\Windows\CurrentVersion\Run", "CurrentUser"),
        (winreg.HKEY_LOCAL_MACHINE, r"Software\Microsoft\Windows\CurrentVersion\Run", "AllUsers")
    ]

    for root, path, user in reg_paths:
        try:
            key = winreg.OpenKey(root, path)
            i = 0
            while True:
                try:
                    name, value, _ = winreg.EnumValue(key, i)
                    items.append({
                        "type": "registry",
                        "user": user,
                        "name": name,
                        "path": value
                    })
                    i += 1
                except OSError:
                    break
        except FileNotFoundError:
            continue

    return items

def disable_startup_item(item):
    """禁用启动项（移动文件/删除注册表值）"""
    if item["type"] == "folder":
        # 移动到 Startup.disabled
        disabled_dir = os.path.join(os.path.dirname(item["path"]), "Startup.disabled")
        os.makedirs(disabled_dir, exist_ok=True)
        shutil.move(item["path"], os.path.join(disabled_dir, item["name"]))
        print(f"已禁用启动项: {item['name']} (文件夹方式)")
    elif item["type"] == "registry":
        if item["user"] == "CurrentUser":
            root = winreg.HKEY_CURRENT_USER
        else:
            root = winreg.HKEY_LOCAL_MACHINE
        path = r"Software\Microsoft\Windows\CurrentVersion\Run"
        try:
            key = winreg.OpenKey(root, path, 0, winreg.KEY_SET_VALUE)
            winreg.DeleteValue(key, item["name"])
            winreg.CloseKey(key)
            print(f"已禁用启动项: {item['name']} (注册表方式)")
        except Exception as e:
            print(f"禁用失败: {e}")


if __name__ == "__main__":
    print("==== 启动项列表 ====")
    items = list_startup_items()
    for idx, item in enumerate(items):
        print(f"[{idx}] {item['user']} - {item['type']} - {item['name']} -> {item['path']}")

    choice = input("输入要禁用的编号 (或直接回车跳过): ").strip()
    if choice.isdigit():
        idx = int(choice)
        if 0 <= idx < len(items):
            disable_startup_item(items[idx])

import sys
import os
import struct
import datetime
import winreg
from PyQt5.QtWidgets import (
    QApplication, QWidget, QVBoxLayout, QTableWidget, QTableWidgetItem,
    QPushButton, QHBoxLayout, QMessageBox, QHeaderView
)

# --------- 工具函数 ---------
def get_startup_folder(user=True):
    base = os.path.expandvars(r"%APPDATA%\Microsoft\Windows\Start Menu\Programs") if user \
        else os.path.expandvars(r"%ProgramData%\Microsoft\Windows\Start Menu\Programs")
    return os.path.join(base, "Startup")

def read_reg_values(root, path):
    items = {}
    try:
        key = winreg.OpenKey(root, path)
    except FileNotFoundError:
        return items
    i = 0
    while True:
        try:
            name, value, vtype = winreg.EnumValue(key, i)
            items[name] = (value, vtype)
            i += 1
        except OSError:
            break
    winreg.CloseKey(key)
    return items

def read_approved_state(root, path, name):
    """返回 'enabled' / 'disabled' / 'unknown'"""
    try:
        key = winreg.OpenKey(root, path)
        data, vtype = winreg.QueryValueEx(key, name)
        winreg.CloseKey(key)
        if not isinstance(data, (bytes, bytearray)) or len(data) < 4:
            return 'unknown'
        code = int.from_bytes(data[:4], 'little')
        if code in (2, 6):
            return 'enabled'
        elif code == 3:
            return 'disabled'
        else:
            return 'unknown'
    except FileNotFoundError:
        return 'unknown'
    except OSError:
        return 'unknown'

def filetime_now_bytes():
    # FILETIME: 100ns ticks since 1601-01-01 UTC
    EPOCH = datetime.datetime(1601, 1, 1, tzinfo=datetime.timezone.utc)
    now = datetime.datetime.now(datetime.timezone.utc)
    ticks = int((now - EPOCH).total_seconds() * 10_000_000)
    return struct.pack("<Q", ticks)

def write_approved_state(root, path, name, enable=True):
    try:
        key = winreg.CreateKey(root, path)
        if enable:
            data = b"\x02\x00\x00\x00" + (b"\x00" * 8)  # 02 + 零FILETIME
        else:
            data = b"\x03\x00\x00\x00" + filetime_now_bytes()  # 03 + 禁用时间
        winreg.SetValueEx(key, name, 0, winreg.REG_BINARY, data)
        winreg.CloseKey(key)
        return True, None
    except PermissionError as e:
        return False, "需要管理员权限 (UAC)。"
    except Exception as e:
        return False, str(e)

# --------- 数据模型构建 ---------
def enumerate_startup_items():
    """
    仅从“真实来源”取项：Run/Run32/启动文件夹
    然后用 StartupApproved 计算“最新状态”
    """
    items = []

    # 1) 注册表 Run（当前用户 / 所有用户 64位 / 所有用户 32位）
    reg_sources = [
        {
            "scope": "CurrentUser",
            "root": winreg.HKEY_CURRENT_USER,
            "run_path": r"Software\Microsoft\Windows\CurrentVersion\Run",
            "approved_path": r"Software\Microsoft\Windows\CurrentVersion\Explorer\StartupApproved\Run",
            "source": "Run"
        },
        {
            "scope": "AllUsers",
            "root": winreg.HKEY_LOCAL_MACHINE,
            "run_path": r"Software\Microsoft\Windows\CurrentVersion\Run",
            "approved_path": r"Software\Microsoft\Windows\CurrentVersion\Explorer\StartupApproved\Run",
            "source": "Run"
        },
        {
            "scope": "AllUsers",
            "root": winreg.HKEY_LOCAL_MACHINE,
            "run_path": r"Software\WOW6432Node\Microsoft\Windows\CurrentVersion\Run",
            "approved_path": r"Software\Microsoft\Windows\CurrentVersion\Explorer\StartupApproved\Run32",
            "source": "Run32"
        },
    ]

    for src in reg_sources:
        values = read_reg_values(src["root"], src["run_path"])
        for name, (cmd, vtype) in values.items():
            state = read_approved_state(src["root"], src["approved_path"], name)
            # 未设置 StartupApproved 时，Windows 默认认为启用
            effective = "启用" if state in ("enabled", "unknown") else "已禁用"
            items.append({
                "scope": src["scope"],
                "type": "registry",
                "source": src["source"],
                "name": name,
                "value": str(cmd),
                "approved_state": state,
                "effective": effective,
                "root": src["root"],
                "run_path": src["run_path"],
                "approved_path": src["approved_path"]
            })

    # 2) 启动文件夹（当前用户 / 所有用户）
    folder_sources = [
        ("CurrentUser", True, winreg.HKEY_CURRENT_USER, r"Software\Microsoft\Windows\CurrentVersion\Explorer\StartupApproved\StartupFolder"),
        ("AllUsers", False, winreg.HKEY_LOCAL_MACHINE, r"Software\Microsoft\Windows\CurrentVersion\Explorer\StartupApproved\StartupFolder"),
    ]
    for scope, is_user, approved_root, approved_path in folder_sources:
        folder = get_startup_folder(user=is_user)
        if os.path.isdir(folder):
            for fn in os.listdir(folder):
                full = os.path.join(folder, fn)
                if not os.path.isfile(full):
                    continue
                # 任务管理器里 StartupApproved\StartupFolder 的值名通常是文件名（含扩展名）
                name = fn
                state = read_approved_state(approved_root, approved_path, name)
                effective = "启用" if state in ("enabled", "unknown") else "已禁用"
                items.append({
                    "scope": scope,
                    "type": "folder",
                    "source": "StartupFolder",
                    "name": name,
                    "value": full,
                    "approved_state": state,
                    "effective": effective,
                    "approved_root": approved_root,
                    "approved_path": approved_path
                })

    return items

def change_item_state(item, enable=True):
    if item["type"] == "registry":
        # 写 StartupApproved 对应项
        ok, err = write_approved_state(item["root"], item["approved_path"], item["name"], enable)
        return ok, err
    else:  # folder
        ok, err = write_approved_state(item["approved_root"], item["approved_path"], item["name"], enable)
        return ok, err

# --------- Qt UI ---------
class StartupManager(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("启动项管理器（精准状态 | 基于 StartupApproved）")
        self.resize(980, 520)

        layout = QVBoxLayout(self)

        self.table = QTableWidget()
        self.table.setColumnCount(7)
        self.table.setHorizontalHeaderLabels(["作用域", "来源", "类型", "名称", "命令/路径", "状态(生效)", "原始状态"])
        self.table.horizontalHeader().setSectionResizeMode(QHeaderView.ResizeMode.Interactive)
        self.table.horizontalHeader().setStretchLastSection(True)
        self.table.setSortingEnabled(True)
        layout.addWidget(self.table)

        row = QHBoxLayout()
        self.btn_refresh = QPushButton("刷新")
        self.btn_enable = QPushButton("启用")
        self.btn_disable = QPushButton("禁用")
        row.addWidget(self.btn_refresh)
        row.addWidget(self.btn_enable)
        row.addWidget(self.btn_disable)
        layout.addLayout(row)

        self.btn_refresh.clicked.connect(self.load_items)
        self.btn_enable.clicked.connect(self.enable_selected)
        self.btn_disable.clicked.connect(self.disable_selected)

        self.load_items()

    def load_items(self):
        self.items = enumerate_startup_items()
        self.table.setRowCount(len(self.items))
        for i, it in enumerate(self.items):
            self.table.setItem(i, 0, QTableWidgetItem(it["scope"]))
            self.table.setItem(i, 1, QTableWidgetItem(it["source"]))
            self.table.setItem(i, 2, QTableWidgetItem(it["type"]))
            self.table.setItem(i, 3, QTableWidgetItem(it["name"]))
            self.table.setItem(i, 4, QTableWidgetItem(it["value"]))
            self.table.setItem(i, 5, QTableWidgetItem(it["effective"]))  # 真实生效状态
            # 原始 StartupApproved 原码状态，便于排查
            raw = {"enabled": "启用(Approved)", "disabled": "禁用(Approved)", "unknown": "未知(未设置)"}.get(it["approved_state"], "未知")
            self.table.setItem(i, 6, QTableWidgetItem(raw))

        # 列宽优化
        self.table.resizeColumnsToContents()
        self.table.setColumnWidth(4, 520)

    def current_item(self):
        r = self.table.currentRow()
        if r < 0:
            QMessageBox.information(self, "提示", "请先选择一条启动项")
            return None
        return self.items[r]

    def enable_selected(self):
        it = self.current_item()
        if not it: return
        ok, err = change_item_state(it, True)
        if ok:
            QMessageBox.information(self, "成功", f"已启用：{it['name']}")
            self.load_items()
        else:
            QMessageBox.critical(self, "失败", err or "启用失败")

    def disable_selected(self):
        it = self.current_item()
        if not it: return
        ok, err = change_item_state(it, False)
        if ok:
            QMessageBox.information(self, "成功", f"已禁用：{it['name']}")
            self.load_items()
        else:
            QMessageBox.critical(self, "失败", err or "禁用失败")

if __name__ == "__main__":
    app = QApplication(sys.argv)
    w = StartupManager()
    w.show()
    sys.exit(app.exec_())

“状态(生效)”= 根据 StartupApproved 计算的最终生效状态（与任务管理器一致）。

“原始状态”= 直接读取 StartupApproved 的原始标志，便于排查。

只枚举 Run / Run32 / 启动文件夹 三类常见来源；其它持久化方式（计划任务、服务、Policies\Run 等）可按需再扩展。

修改 AllUsers(HKLM) 时需要管理员权限（UAC）。

import ctypes
import winreg
import sys

def is_admin():
    """判断是否为管理员权限"""
    try:
        return ctypes.windll.shell32.IsUserAnAdmin()
    except:
        return False

def get_login_state():
    """
    获取 Windows 登录状态
    - 返回 True 表示在登录界面
    - False 表示已登录
    """
    WTSGetActiveConsoleSessionId = ctypes.windll.kernel32.WTSGetActiveConsoleSessionId
    session_id = WTSGetActiveConsoleSessionId()
    return session_id == 0xFFFFFFFF

def clear_login_wallpaper():
    """
    清空 Windows 登录界面壁纸
    设置为纯色（黑色）
    """
    key_path = r"SOFTWARE\Policies\Microsoft\Windows\System"
    try:
        reg = winreg.ConnectRegistry(None, winreg.HKEY_LOCAL_MACHINE)
        key = winreg.CreateKeyEx(reg, key_path, 0, winreg.KEY_WRITE)
        winreg.SetValueEx(key, "DisableLogonBackgroundImage", 0, winreg.REG_DWORD, 1)
        winreg.CloseKey(key)
        print("[+] 已清空登录界面壁纸")
    except Exception as e:
        print("[-] 清空壁纸失败:", e)

def disable_login_ads():
    """
    关闭 Windows 登录界面的广告/小提示
    """
    paths = [
        r"SOFTWARE\Microsoft\Windows\CurrentVersion\ContentDeliveryManager",
        r"SOFTWARE\Policies\Microsoft\Windows\CloudContent"
    ]
    try:
        for path in paths:
            reg = winreg.ConnectRegistry(None, winreg.HKEY_LOCAL_MACHINE)
            key = winreg.CreateKeyEx(reg, path, 0, winreg.KEY_WRITE)
            winreg.SetValueEx(key, "DisableWindowsSpotlightFeatures", 0, winreg.REG_DWORD, 1)
            winreg.SetValueEx(key, "DisableConsumerFeatures", 0, winreg.REG_DWORD, 1)
            winreg.CloseKey(key)
        print("[+] 已关闭登录界面广告与推荐")
    except Exception as e:
        print("[-] 关闭广告失败:", e)
