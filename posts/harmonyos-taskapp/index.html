// ä»»åŠ¡æ•°æ®æ¨¡å‹
export class Task {
  id: string = '';
  title: string = '';
  type: TaskType = TaskType.TIME_BASED;
  status: TaskStatus = TaskStatus.IN_PROGRESS;
  
  // æ—¶é—´ç±»ä»»åŠ¡å­—æ®µ
  startDate?: Date;
  endDate?: Date;
  
  // å­˜é’±ç±»ä»»åŠ¡å­—æ®µ
  targetAmount?: number;
  currentAmount?: number;
  
  createdAt: Date = new Date();
  updatedAt: Date = new Date();
  
  constructor() {
    this.id = this.generateId();
  }
  
  private generateId(): string {
    return Date.now().toString(36) + Math.random().toString(36).substring(2);
  }
  
  // è®¡ç®—æ—¶é—´ç±»ä»»åŠ¡çš„è¿›åº¦ç™¾åˆ†æ¯”
  getTimeProgress(): number {
    if (this.type !== TaskType.TIME_BASED || !this.startDate || !this.endDate) {
      return 0;
    }
    
    const now = new Date();
    const totalTime = this.endDate.getTime() - this.startDate.getTime();
    const elapsedTime = now.getTime() - this.startDate.getTime();
    
    if (elapsedTime <= 0) return 0;
    if (elapsedTime >= totalTime) return 100;
    
    return Math.round((elapsedTime / totalTime) * 100);
  }
  
  // è®¡ç®—å­˜é’±ç±»ä»»åŠ¡çš„è¿›åº¦ç™¾åˆ†æ¯”
  getMoneyProgress(): number {
    if (this.type !== TaskType.MONEY_BASED || !this.targetAmount || this.targetAmount <= 0) {
      return 0;
    }
    
    const current = this.currentAmount || 0;
    const progress = (current / this.targetAmount) * 100;
    return Math.min(Math.round(progress), 100);
  }
  
  // è·å–ä»»åŠ¡è¿›åº¦ç™¾åˆ†æ¯”
  getProgress(): number {
    return this.type === TaskType.TIME_BASED ? this.getTimeProgress() : this.getMoneyProgress();
  }
  
  // è®¡ç®—å‰©ä½™å¤©æ•°
  getRemainingDays(): number {
    if (this.type !== TaskType.TIME_BASED || !this.endDate) {
      return 0;
    }
    
    const now = new Date();
    const remaining = this.endDate.getTime() - now.getTime();
    return Math.ceil(remaining / (1000 * 60 * 60 * 24));
  }
  
  // è®¡ç®—å‰©ä½™é‡‘é¢
  getRemainingAmount(): number {
    if (this.type !== TaskType.MONEY_BASED || !this.targetAmount) {
      return 0;
    }
    
    const current = this.currentAmount || 0;
    return Math.max(this.targetAmount - current, 0);
  }
  
  // æ›´æ–°ä»»åŠ¡çŠ¶æ€
  updateStatus(): void {
    if (this.type === TaskType.TIME_BASED) {
      const progress = this.getTimeProgress();
      const remainingDays = this.getRemainingDays();
      
      if (progress >= 100 || remainingDays <= 0) {
        this.status = TaskStatus.COMPLETED;
      } else if (remainingDays < 0) {
        this.status = TaskStatus.OVERDUE;
      } else {
        this.status = TaskStatus.IN_PROGRESS;
      }
    } else {
      const progress = this.getMoneyProgress();
      this.status = progress >= 100 ? TaskStatus.COMPLETED : TaskStatus.IN_PROGRESS;
    }
    
    this.updatedAt = new Date();
  }
}
