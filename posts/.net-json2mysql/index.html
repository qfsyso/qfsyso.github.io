SQL

CREATE TABLE CityInfo (
CtId INT NOT NULL AUTO_INCREMENT,
Pid INT DEFAULT '0',
CityCode VARCHAR(20) DEFAULT '',
CityName VARCHAR(110) DEFAULT '',
PostCode VARCHAR(20) DEFAULT '',
AreaCode VARCHAR(20) DEFAULT '',
Ctime DATETIME DEFAULT NULL,
CtimeInt bigint DEFAULT '0',
LangType INT DEFAULT '0',
PRIMARY KEY (CtId)
);

    public string AreaCode { get; set; }
    public DateTime? Ctime { get; set; }
    public long CtimeInt { get; set; }
    public int LangType { get; set; }
}

后端处理 json file »> class »> sql

  [HttpPost]
    public ActionResult ImportCities(IFormFile file)
    {
        try
        {
            if (file == null || file.Length == 0)
            {
                return Json(new { success = false, message = "No file uploaded." });
            }

            using (var streamReader = new StreamReader(file.InputStream))
            {
                var jsonData = streamReader.ReadToEnd();
                var cityList = JsonConvert.DeserializeObject<List<CityInfo>>(jsonData);

                using (MySqlConnection connection = new MySqlConnection(ConnectionString))
                {
                    connection.Open();

                    foreach (var city in cityList)
                    {
                        string query = "INSERT INTO IMCityInfo (Pid, CityCode, CityName, PostCode, AreaCode, Ctime, CtimeInt, LangType) VALUES (@Pid, @CityCode, @CityName, @PostCode, @AreaCode, @Ctime, @CtimeInt, @LangType)";

                        using (MySqlCommand command = new MySqlCommand(query, connection))
                        {
                            command.Parameters.AddWithValue("@Pid", city.Pid);
                            command.Parameters.AddWithValue("@CityCode", city.CityCode);
                            command.Parameters.AddWithValue("@CityName", city.CityName);
                            command.Parameters.AddWithValue("@PostCode", city.PostCode);
                            command.Parameters.AddWithValue("@AreaCode", city.AreaCode);
                            command.Parameters.AddWithValue("@Ctime", city.Ctime);
                            command.Parameters.AddWithValue("@CtimeInt", city.CtimeInt);
                            command.Parameters.AddWithValue("@LangType", city.LangType);

                            command.ExecuteNonQuery();
                        }
                    }
                }
            }

            return Json(new { success = true, message = "Data imported successfully." });
        }
        catch (Exception ex)
        {
            return Json(new { success = false, message = ex.Message });
        }
    }

$(document).ready(function () {
    $('#uploadForm').on('submit', function (event) {
        event.preventDefault(); // Prevent the default form submission

        var formData = new FormData();
        var fileInput = $('#jsonFile')[0];
        
        if (fileInput.files.length > 0) {
            var file = fileInput.files[0];
            formData.append('file', file);

            $.ajax({
                url: '/ImportCities', // Adjust this URL to match your API endpoint
                type: 'POST',
                data: formData,
                processData: false, // Important!
                contentType: false, // Important!
                success: function (response) {
                    $('#response').text('Success: ' + response.message);
                },
                error: function (error) {
                    $('#response').text('Error: ' + error.responseJSON.error);
                }
            });
        } else {
            $('#response').text('Please select a file.');
        }
    });
});
